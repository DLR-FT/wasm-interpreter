210pub fn fc_extension_opcode_to_str(instr: u32) -> alloc::borrow::Cow<'static, str> {
211    match instr {
212        0x00 => "I32_TRUNC_SAT_F32_S",
213        0x01 => "I32_TRUNC_SAT_F32_U",
214        0x02 => "I32_TRUNC_SAT_F64_S",
215        0x03 => "I32_TRUNC_SAT_F64_U",
216        0x04 => "I64_TRUNC_SAT_F32_S",
217        0x05 => "I64_TRUNC_SAT_F32_U",
218        0x06 => "I64_TRUNC_SAT_F64_S",
219        0x07 => "I64_TRUNC_SAT_F64_U",
220        0x08 => "MEMORY_INIT",
221        0x09 => "DATA_DROP",
222        0x0A => "MEMORY_COPY",
223        0x0B => "MEMORY_FILL",
224        0x0C => "TABLE_INIT",
225        0x0D => "ELEM_DROP",
226        0x0E => "TABLE_COPY",
227        0x0F => "TABLE_GROW",
228        0x10 => "TABLE_SIZE",
229        0x11 => "TABLE_FILL",
230        instr => return alloc::format!("UNKNOWN({instr:x})").into(),
231    }
232    .into()
233}
