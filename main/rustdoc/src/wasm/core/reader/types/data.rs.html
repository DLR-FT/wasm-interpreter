33impl WasmReadable for DataSegment {
34    fn read(wasm: &mut crate::core::reader::WasmReader) -> crate::Result<Self> {
35        use crate::{NumType, ValType};
36        let mode = wasm.read_var_u32()?;
37        let data_sec: DataSegment = match mode {
38            0 => {
39                // active { memory 0, offset e }
40                trace!("Data section: active");
41                let mut valid_stack = ValidationStack::new();
42                let offset =
43                    { read_constant_expression(wasm, &mut valid_stack, None, None, None)? };
44
45                valid_stack.assert_pop_val_type(ValType::NumType(NumType::I32))?;
46
47                let byte_vec = wasm.read_vec(|el| el.read_u8())?;
48
49                // WARN: we currently don't take into consideration how we act when we are dealing with globals here
50                DataSegment {
51                    mode: DataMode::Active(DataModeActive {
52                        memory_idx: 0,
53                        offset,
54                    }),
55                    init: byte_vec,
56                }
57            }
58            1 => {
59                // passive
60                // A passive data segment's contents can be copied into a memory using the `memory.init` instruction
61                trace!("Data section: passive");
62                DataSegment {
63                    mode: DataMode::Passive,
64                    init: wasm.read_vec(|el| el.read_u8())?,
65                }
66            }
67            2 => {
68                // mode active { memory x, offset e }
69                // this hasn't been yet implemented in wasm
70                // as per docs:
71
72                // https://webassembly.github.io/spec/core/binary/modules.html#data-section
73                // The initial integer can be interpreted as a bitfield. Bit 0 indicates a passive segment, bit 1 indicates the presence of an explicit memory index for an active segment.
74                // In the current version of WebAssembly, at most one memory may be defined or imported in a single module, so all valid active data segments have a memory value of 0
75                todo!("Data section: active - with multiple memories - NOT YET IMPLEMENTED!");
76            }
77            _ => unreachable!(),
78        };
79
80        trace!("{:?}", data_sec.init);
81        Ok(data_sec)
82    }

84    fn read_unvalidated(wasm: &mut crate::core::reader::WasmReader) -> Self {
85        let mode = wasm.read_var_u32().unwrap_validated();
86        let data_sec: DataSegment = match mode {
87            0 => {

98
99                // WARN: we currently don't take into consideration how we act when we are dealing with globals here
100                DataSegment {
101                    mode: DataMode::Active(DataModeActive {
102                        memory_idx: 0,
103                        offset,
104                    }),
105                    init: byte_vec,
106                }
107            }
108            1 => {
109                // passive
110                // A passive data segment's contents can be copied into a memory using the `memory.init` instruction
111                trace!("Data section: passive");
112                DataSegment {
113                    mode: DataMode::Passive,
114                    init: wasm
115                        .read_vec(|el| Ok(el.read_u8().unwrap_validated()))
116                        .unwrap_validated(),
117                }
118            }
119            2 => {
120                // mode active { memory x, offset e }
121                // this hasn't been yet implemented in wasm
122                // as per docs:
123
124                // https://webassembly.github.io/spec/core/binary/modules.html#data-section
125                // The initial integer can be interpreted as a bitfield. Bit 0 indicates a passive segment, bit 1 indicates the presence of an explicit memory index for an active segment.
126                // In the current version of WebAssembly, at most one memory may be defined or imported in a single module, so all valid active data segments have a memory value of 0
127                todo!("Data section: active - with multiple memories - NOT YET IMPLEMENTED!");
128            }
129            _ => unreachable!(),
130        };

132        data_sec
133    }

136impl Debug for DataSegment {
137    fn fmt(&self, f: &mut Formatter<'_>) -> core::fmt::Result {
138        let mut init_str = alloc::string::String::new();
139
140        let iter = self.init.iter().peekable();
141        // only if it's valid do we print is as a normal utf-8 char, otherwise, hex
142        for &byte in iter {
143            if let Ok(valid_char) = alloc::string::String::from_utf8(Vec::from(&[byte])) {
144                init_str.push_str(valid_char.as_str());
145            } else {
146                init_str.push_str(&format!("\\x{:02x}", byte));
147            }
148        }
149
150        f.debug_struct("DataSegment")
151            .field("init", &init_str)
152            .field("mode", &self.mode)
153            .finish()
154    }
155}

157///
158///  Usually, we'd have something like this:
159/// ```wasm
160/// (module
161///  (memory 1) ;; memory starting with 1 page
162///  (data (i32.const 0) "abc")  ;; writing the array of byte "abc" in the first memory (0) at offset 0
163///                             ;; for hardcoded offsets, we'll usually use i32.const because of wasm being x86 arch
164/// )
165/// ```
166///
167/// Since the span has only the start and length and acts a reference, we print the start and end (both inclusive, notice the '..=')
168/// We print it in both decimal and hexadecimal so it's easy to trace in something like <https://webassembly.github.io/wabt/demo/wat2wasm/>
169impl Debug for DataMode {
170    fn fmt(&self, f: &mut Formatter<'_>) -> core::fmt::Result {
171        match self {
172            DataMode::Passive => f.debug_struct("Passive").finish(),
173            DataMode::Active(active_data_mode) => {
174                let from = active_data_mode.offset.from;
175                let to = active_data_mode.offset.from + active_data_mode.offset.len() - 1;
176                f.debug_struct("Active")
177                    // .field("offset", format_args!("[{}..={}]", from, to))
178                    .field(
179                        "offset",
180                        &format_args!("[{}..={}] (hex = [{:X}..={:X}])", from, to, from, to),
181                    )
182                    .finish()
183                // f.
184            }
185        }
186    }
187}
