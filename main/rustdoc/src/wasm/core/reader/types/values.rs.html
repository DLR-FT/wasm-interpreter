17//! Methods to read basic WASM Values from a [WasmReader] object.
18//!
19//! See: <https://webassembly.github.io/spec/core/binary/values.html>
20//!
21//! Note: If any of these methods return `Err`, they may have consumed some bytes from the [WasmReader] object and thus consequent calls may result in unexpected behaviour.
22//! This is due to the fact that these methods read elemental types which cannot be split.
23
24use alloc::vec::Vec;
25use core::mem;
26
27use crate::core::reader::WasmReader;
28use crate::ValidationError;

30impl WasmReader<'_> {
31    /// Note: If `Err`, the [WasmReader] object is no longer guaranteed to be in a valid state
32    pub fn read_u8(&mut self) -> Result<u8, ValidationError> {
33        match self.peek_u8() {
34            Err(e) => Err(e),
35            Ok(byte) => {
36                self.pc += 1;
37                Ok(byte)
38            }
39        }
40    }
41
42    const CONTINUATION_BIT: u8 = 1 << 7;
43
44    /// Parses a variable-length `u64` (can be casted to a smaller uint if the result fits)
45    /// Taken from <https://github.com/bytecodealliance/wasm-tools>
46    #[allow(unused)]
47    pub fn read_var_u64(&mut self) -> Result<u64, ValidationError> {
48        let mut result = 0;
49        let mut shift = 0;
50
51        loop {
52            let mut byte = self.read_u8()?;
53            // maximum allowed num of leb bytes for u32 is ceil(32.0/7.0) == 10
54            // shift >= 63 checks we're at the 10th bit or larger
55            // byte != 1 checks whether (this byte lost bits when shifted) or (the continuation bit is set)
56            if shift == 63 && byte != 0 && byte != 1 {
57                while byte & Self::CONTINUATION_BIT != 0 {
58                    byte = self.read_u8()?;
59                }
60                return Err(ValidationError::Overflow);
61            }
62
63            let low_bits = (byte & !Self::CONTINUATION_BIT) as u64;
64            result |= low_bits << shift;
65
66            if byte & Self::CONTINUATION_BIT == 0 {
67                return Ok(result);
68            }
69
70            shift += 7;
71        }
72    }

74    /// Parses a variable-length `u32` as specified by [LEB128](https://en.wikipedia.org/wiki/LEB128#Unsigned_LEB128).
75    /// Note: If `Err`, the [WasmReader] object is no longer guaranteed to be in a valid state
76    pub fn read_var_u32(&mut self) -> Result<u32, ValidationError> {
77        let mut result: u32 = 0;
78        let mut shift = 0;
79
80        loop {
81            let byte = self.read_u8()?;
82            result |= ((byte & 0x7F) as u32) << shift;
83            // maximum allowed num of leb bytes for u32 is ceil(32.0/7.0) == 5
84            // shift >= 28 checks we're at the 5th bit or larger
85            // byte >> 32-28 checks whether (this byte lost bits when shifted) or (the continuation bit is set)
86            if shift >= 28 && byte >> (32 - shift) != 0 {
87                return Err(ValidationError::Overflow);
88            }
89
90            if byte & Self::CONTINUATION_BIT == 0 {
91                return Ok(result);
92            }
93
94            shift += 7;
95        }
96    }

98    pub fn read_var_f64(&mut self) -> Result<u64, ValidationError> {
99        let bytes = self.strip_bytes::<8>().map_err(|_| ValidationError::Eof)?;
100        let word = u64::from_le_bytes(bytes);
101        Ok(word)
102    }

104    /// Adapted from <https://github.com/bytecodealliance/wasm-tools>
105    pub fn read_var_i32(&mut self) -> Result<i32, ValidationError> {
106        let mut result: i32 = 0;
107        let mut shift: u32 = 0;
108
109        loop {
110            let byte = self.read_u8()?;
111            result |= ((byte & 0x7F) as i32) << shift;
112
113            if shift >= 28 {
114                // maximum allowed num of leb bytes for u32 is ceil(32.0/7.0) == 5
115                // shift >= 28 checks we're at the 5th bit or larger
116                let there_are_more_bytes = (byte & Self::CONTINUATION_BIT) != 0;
117                let ashifted_unused_bits = (byte << 1) as i8 >> (32 - shift);
118                // the top unused bits of 35 bytes should be either 111 for negative numbers or 000 for positive numbers
119                // therefore ashifted_unused_bits should be -1 or 0
120                if there_are_more_bytes || (ashifted_unused_bits != 0 && ashifted_unused_bits != -1)
121                {
122                    return Err(ValidationError::Overflow);
123                } else {
124                    // no need to ashift unfilled bits, all 32 bits are filled
125                    return Ok(result);
126                }
127            }
128
129            shift += 7;
130
131            if (byte & 0b10000000) == 0 {
132                break;
133            }
134        }
135
136        // fill in unfilled bits with sign bit
137        let ashift = mem::size_of::<i32>() * 8 - shift as usize;
138        Ok((result << ashift) >> ashift)
139    }

174    pub fn read_var_f32(&mut self) -> Result<u32, ValidationError> {
175        if self.full_wasm_binary.len() - self.pc < 4 {
176            return Err(ValidationError::Eof);
177        }
178
179        let word = u32::from_le_bytes(
180            self.full_wasm_binary[self.pc..self.pc + 4]
181                .try_into()
182                .unwrap(),
183        );
184
185        self.strip_bytes::<4>()?;
186
187        Ok(word)
188    }

190    pub fn read_var_i64(&mut self) -> Result<i64, ValidationError> {
191        let mut result: i64 = 0;
192        let mut shift: u32 = 0;
193
194        loop {
195            let byte = self.read_u8()?;
196            result |= ((byte & 0x7F) as i64) << shift;
197
198            if shift >= 63 {
199                // maximum allowed num of leb bytes for i33 is ceil(64.0/7.0) == 10
200                // shift >= 63 checks we're at the 10th bit or larger
201                let there_are_more_bytes = (byte & Self::CONTINUATION_BIT) != 0;
202                let ashifted_unused_bits = (byte << 1) as i8 >> (64 - shift);
203                // the top unused bits of 70 bytes should be either 111111 for negative numbers or 000000 for positive numbers
204                // therefore ashifted_unused_bits should be -1 or 0
205                if there_are_more_bytes || (ashifted_unused_bits != 0 && ashifted_unused_bits != -1)
206                {
207                    return Err(ValidationError::Overflow);
208                } else {
209                    // no need to ashift unfilled bits, all 64 bits are filled
210                    return Ok(result);
211                }
212            }
213
214            shift += 7;
215
216            if (byte & 0b10000000) == 0 {
217                break;
218            }
219        }
220
221        // fill in unfilled bits with sign bit
222        let ashift = mem::size_of::<i64>() * 8 - shift as usize;
223        Ok((result << ashift) >> ashift)
224    }

226    /// Note: If `Err`, the [WasmReader] object is no longer guaranteed to be in a valid state
227    pub fn read_name(&mut self) -> Result<&str, ValidationError> {
228        let len = self.read_var_u32()? as usize;
229
230        if len > self.full_wasm_binary.len() - self.pc {
231            return Err(ValidationError::Eof);
232        }
233
234        let utf8_str = &self.full_wasm_binary[self.pc..(self.pc + len)]; // Cannot panic because check is done above
235        self.pc += len;
236
237        core::str::from_utf8(utf8_str).map_err(ValidationError::MalformedUtf8String)
238    }
239
240    pub fn read_vec_enumerated<T, F>(
241        &mut self,
242        mut read_element: F,
243    ) -> Result<Vec<T>, ValidationError>
244    where
245        F: FnMut(&mut WasmReader, usize) -> Result<T, ValidationError>,
246    {
247        let mut idx = 0;
248        self.read_vec(|wasm| {
249            let ret = read_element(wasm, idx);
250            idx += 1;
251            ret
252        })
253    }
254
255    /// Note: If `Err`, the [WasmReader] object is no longer guaranteed to be in a valid state
256    pub fn read_vec<T, F>(&mut self, mut read_element: F) -> Result<Vec<T>, ValidationError>
257    where
258        F: FnMut(&mut WasmReader) -> Result<T, ValidationError>,
259    {
260        let len = self.read_var_u32()?;
261        (0..len).map(|_| read_element(self)).collect()
262    }
