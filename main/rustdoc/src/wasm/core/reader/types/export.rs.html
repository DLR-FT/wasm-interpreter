28#[derive(Debug, Clone)]
29pub enum ExportDesc {
30    Func(FuncIdx),
31    Table(TableIdx),
32    Mem(MemIdx),
33    Global(GlobalIdx),
34}
35
36impl ExportDesc {
37    /// returns the external type of `self` according to typing relation,
38    /// taking `validation_info` as validation context C
39    ///
40    /// # Safety
41    ///
42    /// The caller must ensure that `self` comes from the same
43    /// [`ValidationInfo`] that is passed as an argument here.
44    #[allow(unused)] // reason = "this function is analogous to ImportDesc::extern_type, however it is not yet clear if it is needed in the future"
45    pub unsafe fn extern_type(&self, validation_info: &ValidationInfo) -> ExternType {
46        // TODO clean up logic for checking if an exported definition is an
47        // import
48        match self {
49            ExportDesc::Func(func_idx) => {
50                // SAFETY: The caller ensures that the current `ExportDesc`
51                // comes from the same `ValidationInfo` that is passed into the
52                // current function. Therefore, the function index stored in
53                // `self` must be valid in the given `ValidationInfo`.
54                let type_idx = unsafe { validation_info.functions.inner().get(*func_idx) };
55                // SAFETY: The type index was just read from the passed
56                // `ValidationInfo`.  Because the `ValidationInfo` struct
57                // guarantees that all indices contained in it are valid for all
58                // other `IdxVec` vectors in it, this is sound.
59                let func_type = unsafe { validation_info.types.get(*type_idx) };
60                // TODO ugly clone that should disappear when types are directly parsed from bytecode instead of vector copies
61                ExternType::Func(func_type.clone())
62            }
63            ExportDesc::Table(table_idx) => {
64                // SAFETY: The caller ensures that the current `ExportDesc`
65                // comes from the same `ValidationInfo` that is passed into the
66                // current function. Therefore, the table index stored in `self`
67                // must be valid in the given `ValidationInfo`.
68                let table_type = unsafe { validation_info.tables.inner().get(*table_idx) };
69
70                ExternType::Table(*table_type)
71            }
72            ExportDesc::Mem(mem_idx) => {
73                // SAFETY: The caller ensures that the current `ExportDesc`
74                // comes from the same `ValidationInfo` that is passed into the
75                // current function. Therefore, the memory index stored in
76                // `self` must be valid in the given `ValidationInfo`.
77                let mem_type = unsafe { validation_info.memories.inner().get(*mem_idx) };
78
79                ExternType::Mem(*mem_type)
80            }
81            ExportDesc::Global(global_idx) => {
82                // SAFETY: The caller ensures that the current `ExportDesc`
83                // comes from the same `ValidationInfo` that is passed into the
84                // current function. Therefore, the global index stored in
85                // `self` must be valid in the given `ValidationInfo`.
86                let global = unsafe { validation_info.globals.inner().get(*global_idx) };
87
88                ExternType::Global(global.ty)
89            }
90        }
91    }
92}
93
94impl ExportDesc {
95    pub fn read_and_validate(
96        wasm: &mut WasmReader,
97        c_functions: &IdxVec<FuncIdx, TypeIdx>,
98        c_tables: &IdxVec<TableIdx, TableType>,
99        c_mems: &IdxVec<MemIdx, MemType>,
100        c_globals: &IdxVec<GlobalIdx, Global>,
101    ) -> Result<Self, ValidationError> {
102        let desc_id = wasm.read_u8()?;
103
104        let desc = match desc_id {
105            0x00 => ExportDesc::Func(FuncIdx::read_and_validate(wasm, c_functions)?),
106            0x01 => ExportDesc::Table(TableIdx::read_and_validate(wasm, c_tables)?),
107            0x02 => ExportDesc::Mem(MemIdx::read_and_validate(wasm, c_mems)?),
108            0x03 => ExportDesc::Global(GlobalIdx::read_and_validate(wasm, c_globals)?),
109            other => return Err(ValidationError::MalformedExportDescDiscriminator(other)),
110        };
111        Ok(desc)
112    }
