9/// Execute a previosly-validated constant expression. These type of expressions are used for initializing global
10/// variables.
11///
12/// # Arguments
13/// - `wasm` - a [WasmReader] whose [program counter](WasmReader::pc) is set at the beginning of the constant
14///   expression. Reader will be consumed.
15/// - `stack` - a [Stack]. It is preferrable for it to be clean, but that is not required. As long as the executed code
16///   is validated, the values on this stack will remain the same except for the addition of the return value of this
17///   code sequence. A global's final value can be popped off the top of the stack.
18/// - `imported_globals` (TODO) - instances of all imported globals. They are required as local globals can reference
19///   imported globals in their initialization.
20///
21/// # Safety
22/// This function assumes that the expression has been validated. Passing unvalidated code will likely result in a
23/// panic, or undefined behaviour.
24///
25/// # Note
26/// The following instructions are not yet supported:
27/// - `ref.null`
28/// - `ref.func`
29/// - `global.get`
30pub(crate) fn run_const(
31    mut wasm: WasmReader,
32    stack: &mut Stack,
33    _imported_globals: (), /*todo!*/
34) {
35    use crate::core::reader::types::opcode::*;
36    loop {
37        let first_instr_byte = wasm.read_u8().unwrap_validated();
38
39        match first_instr_byte {
40            END => {
41                break;
42            }
43            I32_CONST => {
44                let constant = wasm.read_var_i32().unwrap_validated();
45                trace!("Constant instruction: i32.const [] -> [{constant}]");
46                stack.push_value(constant.into());
47            }
48            F32_CONST => {
49                let constant = value::F32::from_bits(wasm.read_var_f32().unwrap_validated());
50                trace!("Constanting instruction: f32.const [] -> [{constant}]");
51                stack.push_value(constant.into());
52            }
53            F64_CONST => {
54                let constant = value::F64::from_bits(wasm.read_var_f64().unwrap_validated());
55                trace!("Constanting instruction: f64.const [] -> [{constant}]");
56                stack.push_value(constant.into());
57            }
58            I32_ADD => {
59                let v1: i32 = stack.pop_value(ValType::NumType(NumType::I32)).into();
60                let v2: i32 = stack.pop_value(ValType::NumType(NumType::I32)).into();
61                let res = v1.wrapping_add(v2);
62
63                trace!("Constant instruction: i32.add [{v1} {v2}] -> [{res}]");
64                stack.push_value(res.into());
65            }
66            I32_SUB => {
67                let v2: i32 = stack.pop_value(ValType::NumType(NumType::I32)).into();
68                let v1: i32 = stack.pop_value(ValType::NumType(NumType::I32)).into();
69                let res = v1.wrapping_sub(v2);
70
71                trace!("Constant instruction: i32.sub [{v1} {v2}] -> [{res}]");
72                stack.push_value(res.into());
73            }
74            I32_MUL => {
75                let v1: i32 = stack.pop_value(ValType::NumType(NumType::I32)).into();
76                let v2: i32 = stack.pop_value(ValType::NumType(NumType::I32)).into();
77                let res = v1.wrapping_mul(v2);
78
79                trace!("Constant instruction: i32.mul [{v1} {v2}] -> [{res}]");
80                stack.push_value(res.into());
81            }
82            I64_CONST => {
83                let constant = wasm.read_var_i64().unwrap_validated();
84                trace!("Constant instruction: i64.const [] -> [{constant}]");
85                stack.push_value(constant.into());
86            }
87            I64_ADD => {
88                let v1: i64 = stack.pop_value(ValType::NumType(NumType::I64)).into();
89                let v2: i64 = stack.pop_value(ValType::NumType(NumType::I64)).into();
90                let res = v1.wrapping_add(v2);
91
92                trace!("Constant instruction: i64.add [{v1} {v2}] -> [{res}]");
93                stack.push_value(res.into());
94            }
95            I64_SUB => {
96                let v2: i64 = stack.pop_value(ValType::NumType(NumType::I64)).into();
97                let v1: i64 = stack.pop_value(ValType::NumType(NumType::I64)).into();
98                let res = v1.wrapping_sub(v2);
99
100                trace!("Constant instruction: i64.sub [{v1} {v2}] -> [{res}]");
101                stack.push_value(res.into());
102            }
103            I64_MUL => {
104                let v1: i64 = stack.pop_value(ValType::NumType(NumType::I64)).into();
105                let v2: i64 = stack.pop_value(ValType::NumType(NumType::I64)).into();
106                let res = v1.wrapping_mul(v2);
107
108                trace!("Constant instruction: i64.mul [{v1} {v2}] -> [{res}]");
109                stack.push_value(res.into());
110            }
111            REF_NULL => {
112                let reftype = RefType::read_unvalidated(&mut wasm);
113
114                stack.push_value(Value::Ref(reftype.to_null_ref()));
115                trace!("Instruction: ref.null '{:?}' -> [{:?}]", reftype, reftype);
116            }
117            REF_FUNC => {
118                // we already checked for the func_idx to be in bounds during validation
119                let func_idx = wasm.read_var_u32().unwrap_validated() as usize;
120                stack.push_value(Value::Ref(Ref::Func(FuncAddr::new(Some(func_idx)))));
121            }
122            other => {
123                panic!("Unknown constant instruction {other:#x}, validation allowed an unimplemented instruction.");
124            }
125        }
126    }
127}

129pub(crate) fn run_const_span(
130    wasm: &[u8],
131    span: &Span,
132    imported_globals: (),
133    // funcs: &[FuncInst],
134) -> Option<Value> {
135    let mut wasm = WasmReader::new(wasm);
136
137    wasm.move_start_to(*span).unwrap_validated();
138
139    let mut stack = Stack::new();
140    run_const(wasm, &mut stack, imported_globals);
141
142    stack.peek_unknown_value()
143}
