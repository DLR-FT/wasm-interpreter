33/// The store represents all global state that can be manipulated by WebAssembly programs. It
34/// consists of the runtime representation of all instances of functions, tables, memories, and
35/// globals, element segments, and data segments that have been allocated during the life time of
36/// the abstract machine.
37/// <https://webassembly.github.io/spec/core/exec/runtime.html#store>
38#[derive(Default, Debug)]
39pub struct Store<'b> {
40    pub functions: Vec<FuncInst>,
41    pub memories: Vec<MemInst>,
42    pub globals: Vec<GlobalInst>,
43    pub data: Vec<DataInst>,
44    pub tables: Vec<TableInst>,
45    pub elements: Vec<ElemInst>,
46    pub modules: Vec<ModuleInst<'b>>,
47    pub module_names: BTreeMap<String, usize>,
48}

50impl<'b> Store<'b> {
51    /// instantiates a validated module with `validation_info` as validation evidence with name `name`
52    /// with the steps in <https://webassembly.github.io/spec/core/exec/modules.html#instantiation>
53    /// this method roughly matches the suggested embedder function`module_instantiate`
54    /// <https://webassembly.github.io/spec/core/appendix/embedding.html#modules>
55    /// except external values for module instantiation are retrieved from `self`.
56    pub fn add_module(
57        &mut self,
58        name: &str,
59        validation_info: &ValidationInfo<'b>,
60    ) -> CustomResult<()> {
61        // instantiation step -1: collect extern_vals, this section basically acts as a linker between modules
62        // best attempt at trying to match the spec implementation in terms of errors
63        debug!("adding module with name {:?}", name);
64        let mut extern_vals = Vec::new();
65
66        for Import {
67            module_name: exporting_module_name,
68            name: import_name,
69            desc: import_desc,
70        } in &validation_info.imports
71        {
72            trace!(
73                "trying to import from exporting module instance named {:?}, the entity with name {:?} with desc: {:?}",
74                exporting_module_name,
75                import_name,
76                import_desc
77            );
78            let import_extern_type = import_desc.extern_type(validation_info)?;
79            let exporting_module = self
80                .modules
81                .get(
82                    *self
83                        .module_names
84                        .get(exporting_module_name)
85                        .ok_or(Error::RuntimeError(RuntimeError::ModuleNotFound))?,
86                )
87                .ok_or(Error::RuntimeError(RuntimeError::ModuleNotFound))?;
88
89            let export_extern_val_candidate = *exporting_module
90                .exports
91                .iter()
92                .find_map(
93                    |ExportInst {
94                         name: export_name,
95                         value: export_extern_val,
96                     }| {
97                        (import_name == export_name).then_some(export_extern_val)
98                    },
99                )
100                .ok_or(Error::UnknownImport)?;
101            trace!("export candidate found: {:?}", export_extern_val_candidate);
102            if !export_extern_val_candidate
103                .extern_type(self)?
104                .is_subtype_of(&import_extern_type)
105            {
106                return Err(Error::InvalidImportType);
107            }
108            trace!("import and export matches. Adding to externvals");
109            extern_vals.push(export_extern_val_candidate)
110        }
111
112        // instantiation: step 5
113        // module_inst_init is unfortunately circularly defined from parts of module_inst that would be defined in step 11, which uses module_inst_init again implicitly.
114        // therefore I am mimicking the reference interpreter code here, I will allocate functions in the store in this step instead of step 11.
115        // https://github.com/WebAssembly/spec/blob/8d6792e3d6709e8d3e90828f9c8468253287f7ed/interpreter/exec/eval.ml#L789
116        let mut module_inst = ModuleInst {
117            types: validation_info.types.clone(),
118            func_addrs: extern_vals.iter().funcs().collect(),
119            table_addrs: Vec::new(),
120            mem_addrs: Vec::new(),
121            global_addrs: extern_vals.iter().globals().collect(),
122            elem_addrs: Vec::new(),
123            data_addrs: Vec::new(),
124            exports: Vec::new(),
125            wasm_bytecode: validation_info.wasm,
126            sidetable: validation_info.sidetable.clone(),
127            name: name.to_owned(),
128        };
129
130        // TODO rewrite this part
131        // <https://webassembly.github.io/spec/core/exec/modules.html#functions>
132        let func_addrs: Vec<usize> = validation_info
133            .functions
134            .iter()
135            .zip(validation_info.func_blocks_stps.iter())
136            .map(|(ty_idx, (span, stp))| {
137                self.alloc_func((*ty_idx, (*span, *stp)), &module_inst, self.modules.len())
138            })
139            .collect();
140
141        module_inst.func_addrs.extend(func_addrs);
142
143        // instantiation: this roughly matches step 6,7,8
144        // validation guarantees these will evaluate without errors.
145        let global_init_vals: Vec<Value> = validation_info
146            .globals
147            .iter()
148            .map(|global| {
149                run_const_span(validation_info.wasm, &global.init_expr, &module_inst, self)
150                    .unwrap_validated()
151            })
152            .collect();
153
154        // instantiation: this roughly matches step 9,10
155        let element_init_ref_lists: Vec<Vec<Ref>> = validation_info
156            .elements
157            .iter()
158            .map(|elem| {
159                match &elem.init {
160                    // shortcut of evaluation of "ref.func <func_idx>; end;"
161                    // validation guarantees corresponding func_idx's existence
162                    ElemItems::RefFuncs(ref_funcs) => ref_funcs
163                        .iter()
164                        .map(|func_idx| {
165                            Ref::Func(FuncAddr {
166                                addr: Some(module_inst.func_addrs[*func_idx as usize]),
167                            })
168                        })
169                        .collect(),
170                    ElemItems::Exprs(_, exprs) => exprs
171                        .iter()
172                        .map(|expr| {
173                            run_const_span(validation_info.wasm, expr, &module_inst, self)
174                                .unwrap_validated()
175                                .into()
176                        })
177                        .collect(),
178                }
179            })
180            .collect();
181
182        // instantiation: step 11 - module allocation (except function allocation - which was made in step 5)
183        // https://webassembly.github.io/spec/core/exec/modules.html#alloc-module
184
185        // allocation: begin
186
187        // allocation: step 1
188        let module = validation_info;
189
190        let extern_vals = extern_vals;
191        let vals = global_init_vals;
192        let ref_lists = element_init_ref_lists;
193
194        // allocation: skip step 2 as it was done in instantiation step 5
195
196        // allocation: step 3-13
197        let table_addrs: Vec<usize> = module
198            .tables
199            .iter()
200            .map(|table_type| {
201                let null_ref = match table_type.et {
202                    RefType::FuncRef => Ref::Func(FuncAddr { addr: None }),
203                    RefType::ExternRef => Ref::Extern(ExternAddr { addr: None }),
204                };
205                self.alloc_table(*table_type, null_ref)
206            })
207            .collect();
208        let mem_addrs: Vec<usize> = module
209            .memories
210            .iter()
211            .map(|mem_type| self.alloc_mem(*mem_type))
212            .collect();
213        let global_addrs: Vec<usize> = module
214            .globals
215            .iter()
216            .zip(vals)
217            .map(
218                |(
219                    Global {
220                        ty: global_type, ..
221                    },
222                    val,
223                )| self.alloc_global(*global_type, val),
224            )
225            .collect();
226        let elem_addrs = module
227            .elements
228            .iter()
229            .zip(ref_lists)
230            .map(|(elem, refs)| self.alloc_elem(elem.ty(), refs))
231            .collect();
232        let data_addrs = module
233            .data
234            .iter()
235            .map(|DataSegment { init: bytes, .. }| self.alloc_data(bytes))
236            .collect();
237
238        // allocation: skip step 14 as it was done in instantiation step 5
239
240        // allocation: step 15,16
241        let mut table_addrs_mod: Vec<usize> = extern_vals.iter().tables().collect();
242        table_addrs_mod.extend(table_addrs);
243
244        let mut mem_addrs_mod: Vec<usize> = extern_vals.iter().mems().collect();
245        mem_addrs_mod.extend(mem_addrs);
246
247        // skipping step 17 partially as it was partially done in instantiation step
248        module_inst.global_addrs.extend(global_addrs);
249
250        // allocation: step 18,19
251        let export_insts = module
252            .exports
253            .iter()
254            .map(|Export { name, desc }| {
255                let value = match desc {
256                    ExportDesc::FuncIdx(func_idx) => {
257                        ExternVal::Func(module_inst.func_addrs[*func_idx])
258                    }
259                    ExportDesc::TableIdx(table_idx) => {
260                        ExternVal::Table(table_addrs_mod[*table_idx])
261                    }
262                    ExportDesc::MemIdx(mem_idx) => ExternVal::Mem(mem_addrs_mod[*mem_idx]),
263                    ExportDesc::GlobalIdx(global_idx) => {
264                        ExternVal::Global(module_inst.global_addrs[*global_idx])
265                    }
266                };
267                ExportInst {
268                    name: String::from(name),
269                    value,
270                }
271            })
272            .collect();
273
274        // allocation: step 20,21 initialize module (except functions and globals due to instantiation step 5, allocation step 14,17)
275        module_inst.table_addrs = table_addrs_mod;
276        module_inst.mem_addrs = mem_addrs_mod;
277        module_inst.elem_addrs = elem_addrs;
278        module_inst.data_addrs = data_addrs;
279        module_inst.exports = export_insts;
280
281        // allocation: end
282
283        // instantiation step 11 end: module_inst properly allocated after this point.
284        // TODO: it is too hard with our codebase to do the following steps without adding the module to the store
285        let current_module_idx = &self.modules.len();
286        self.modules.push(module_inst);
287        self.module_names
288            .insert(String::from(name), *current_module_idx);
289
290        // instantiation: step 12-15
291        // TODO have to stray away from the spec a bit since our codebase does not lend itself well to freely executing instructions by themselves
292        for (
293            i,
294            ElemType {
295                init: elem_items,
296                mode,
297            },
298        ) in validation_info.elements.iter().enumerate()
299        {
300            match mode {
301                ElemMode::Active(ActiveElem {
302                    table_idx: table_idx_i,
303                    init_expr: einstr_i,
304                }) => {
305                    let n = elem_items.len() as i32;
306                    // equivalent to init.len() in spec
307                    // instantiation step 14:
308                    // TODO (for now, we are doing hopefully what is equivalent to it)
309                    // execute:
310                    //   einstr_i
311                    //   i32.const 0
312                    //   i32.const n
313                    //   table.init table_idx_i i
314                    //   elem.drop i
315                    let d: i32 = run_const_span(
316                        validation_info.wasm,
317                        einstr_i,
318                        &self.modules[*current_module_idx],
319                        self,
320                    )
321                    .unwrap_validated()
322                    .into();
323                    let s = 0;
324                    table_init(
325                        &self.modules,
326                        &mut self.tables,
327                        &self.elements,
328                        current_module_idx,
329                        i,
330                        *table_idx_i as usize,
331                        n,
332                        s,
333                        d,
334                    )
335                    .map_err(Error::RuntimeError)?;
336                    elem_drop(&self.modules, &mut self.elements, current_module_idx, i)
337                        .map_err(Error::RuntimeError)?;
338                }
339                ElemMode::Declarative => {
340                    // instantiation step 15:
341                    // TODO (for now, we are doing hopefully what is equivalent to it)
342                    // execute:
343                    //   elem.drop i
344                    elem_drop(&self.modules, &mut self.elements, current_module_idx, i)
345                        .map_err(Error::RuntimeError)?;
346                }
347                ElemMode::Passive => (),
348            }
349        }
350
351        // instantiation: step 16
352        // TODO have to stray away from the spec a bit since our codebase does not lend itself well to freely executing instructions by themselves
353        for (i, DataSegment { init, mode }) in validation_info.data.iter().enumerate() {
354            match mode {
355                crate::core::reader::types::data::DataMode::Active(DataModeActive {
356                    memory_idx,
357                    offset: dinstr_i,
358                }) => {
359                    let n = init.len() as i32;
360                    // assert: mem_idx is 0
361                    if *memory_idx != 0 {
362                        // TODO fix error
363                        return Err(Error::MoreThanOneMemory);
364                    }
365
366                    // TODO (for now, we are doing hopefully what is equivalent to it)
367                    // execute:
368                    //   dinstr_i
369                    //   i32.const 0
370                    //   i32.const n
371                    //   memory.init i
372                    //   data.drop i
373                    let d: i32 = run_const_span(
374                        validation_info.wasm,
375                        dinstr_i,
376                        &self.modules[*current_module_idx],
377                        self,
378                    )
379                    .unwrap_validated()
380                    .into();
381                    let s = 0;
382                    memory_init(
383                        &self.modules,
384                        &mut self.memories,
385                        &self.data,
386                        current_module_idx,
387                        i,
388                        0,
389                        n,
390                        s,
391                        d,
392                    )
393                    .map_err(Error::RuntimeError)?;
394                    data_drop(&self.modules, &mut self.data, current_module_idx, i)
395                        .map_err(Error::RuntimeError)?;
396                }
397                crate::core::reader::types::data::DataMode::Passive => (),
398            }
399        }
400
401        // instantiation: step 17
402        if let Some(func_idx) = validation_info.start {
403            // TODO (for now, we are doing hopefully what is equivalent to it)
404            // execute
405            //   call func_ifx
406            let func_addr = self.modules[*current_module_idx].func_addrs[func_idx];
407            self.invoke_dynamic(func_addr, Vec::new(), &[])
408                .map_err(Error::RuntimeError)?;
409        };
410
411        Ok(())
412    }

415    // TODO refactor the type of func

417    fn alloc_func(
418        &mut self,
419        func: (TypeIdx, (Span, usize)),
420        module_inst: &ModuleInst,
421        module_addr: usize,
422    ) -> usize {
423        let (ty, (span, stp)) = func;
424
425        // TODO rewrite this huge chunk of parsing after generic way to re-parse(?) structs lands
426        let mut wasm_reader = WasmReader::new(module_inst.wasm_bytecode);
427        wasm_reader.move_start_to(span).unwrap_validated();
428
429        let (locals, bytes_read) = wasm_reader
430            .measure_num_read_bytes(crate::code::read_declared_locals)
431            .unwrap_validated();
432
433        let code_expr = wasm_reader
434            .make_span(span.len() - bytes_read)
435            .unwrap_validated();
436
437        // core of the method below
438
439        let func_inst = FuncInst {
440            ty,
441            locals,
442            code_expr,
443            stp,
444            // validation guarantees func_ty_idx exists within module_inst.types
445            // TODO fix clone
446            function_type: module_inst.types[ty].clone(),
447            module_addr,
448        };
449
450        let addr = self.functions.len();
451        self.functions.push(func_inst);
452        addr
453    }

516    pub fn invoke<Param: InteropValueList, Returns: InteropValueList>(
517        &mut self,
518        func_addr: usize,
519        params: Param,
520    ) -> Result<Returns, RuntimeError> {
521        let func_inst = self
522            .functions
523            .get(func_addr)
524            .ok_or(RuntimeError::FunctionNotFound)?;
525
526        let func_ty = func_inst.ty();
527
528        if func_ty.params.valtypes != Param::TYS {
529            panic!("Invalid `Param` generics");
530        }
531        if func_ty.returns.valtypes != Returns::TYS {
532            panic!("Invalid `Returns` generics");
533        }
534
535        let mut stack = Stack::new();
536        let locals = Locals::new(
537            params.into_values().into_iter(),
538            func_inst.locals.iter().cloned(),
539        );
540
541        let module_addr = func_inst.module_addr;
542
543        // TODO handle this bad linear search that is unavoidable
544        let (func_idx, _) = self.modules[module_addr]
545            .func_addrs
546            .iter()
547            .enumerate()
548            .find(|&(_idx, addr)| *addr == func_addr)
549            .ok_or(RuntimeError::FunctionNotFound)?;
550        // setting `usize::MAX` as return address for the outermost function ensures that we
551        // observably fail upon errornoeusly continuing execution after that function returns.
552        stack.push_stackframe(
553            module_addr,
554            func_idx,
555            &func_ty,
556            locals,
557            usize::MAX,
558            usize::MAX,
559        );
560
561        let mut current_module_idx = module_addr;
562        // Run the interpreter
563        run(
564            // &mut self.modules,
565            &mut current_module_idx,
566            // self.lut.as_ref().ok_or(RuntimeError::UnmetImport)?,
567            &mut stack,
568            EmptyHookSet,
569            self,
570        )?;
571
572        // Pop return values from stack
573        let return_values = Returns::TYS
574            .iter()
575            .rev()
576            .map(|ty| stack.pop_value(*ty))
577            .collect::<Vec<Value>>();
578
579        // Values are reversed because they were popped from stack one-by-one. Now reverse them back
580        let reversed_values = return_values.into_iter().rev();
581        let ret: Returns = Returns::from_values(reversed_values);
582        debug!("Successfully invoked function");
583        Ok(ret)
584    }

586    pub fn invoke_dynamic(
587        &mut self,
588        func_addr: usize,
589        params: Vec<Value>,
590        ret_types: &[ValType],
591    ) -> Result<Vec<Value>, RuntimeError> {
592        let func_inst = self
593            .functions
594            .get(func_addr)
595            .ok_or(RuntimeError::FunctionNotFound)?;
596
597        let func_ty = func_inst.ty();
598
599        // Verify that the given parameters match the function parameters
600        let param_types = params.iter().map(|v| v.to_ty()).collect::<Vec<_>>();
601
602        if func_ty.params.valtypes != param_types {
603            // format!()
604            trace!(
605                "Func param types len: {}; Given args len: {}",
606                func_ty.params.valtypes.len(),
607                param_types.len()
608            );
609            panic!("Invalid parameters for function");
610        }
611
612        // Verify that the given return types match the function return types
613        if func_ty.returns.valtypes != ret_types {
614            panic!("Invalid return types for function");
615        }
616
617        // Prepare a new stack with the locals for the entry function
618        let mut stack = Stack::new();
619        let locals = Locals::new(params.into_iter(), func_inst.locals.iter().cloned());
620        let module_addr = func_inst.module_addr;
621
622        // TODO handle this bad linear search that is unavoidable
623        let (func_idx, _) = self.modules[module_addr]
624            .func_addrs
625            .iter()
626            .enumerate()
627            .find(|&(_idx, addr)| *addr == func_addr)
628            .ok_or(RuntimeError::FunctionNotFound)?;
629        stack.push_stackframe(
630            module_addr,
631            func_idx,
632            &func_ty,
633            locals,
634            usize::MAX,
635            usize::MAX,
636        );
637
638        let mut currrent_module_idx = module_addr;
639        // Run the interpreter
640        run(
641            // &mut self.modules,
642            &mut currrent_module_idx,
643            // self.lut.as_ref().ok_or(RuntimeError::UnmetImport)?,
644            &mut stack,
645            EmptyHookSet,
646            self,
647        )?;
648
649        let func_inst = self
650            .functions
651            .get(func_addr)
652            .ok_or(RuntimeError::FunctionNotFound)?;
653
654        let func_ty = func_inst.ty();
655
656        // Pop return values from stack
657        let return_values = func_ty
658            .returns
659            .valtypes
660            .iter()
661            .rev()
662            .map(|ty| stack.pop_value(*ty))
663            .collect::<Vec<Value>>();
664
665        // Values are reversed because they were popped from stack one-by-one. Now reverse them back
666        let reversed_values = return_values.into_iter().rev();
667        let ret = reversed_values.collect();
668        debug!("Successfully invoked function");
669        Ok(ret)
670    }

674        func_addr: usize,

676    ) -> Result<Vec<Value>, RuntimeError> {

688
689        if func_ty.params.valtypes != param_types {
690            trace!(
691                "Func param types len: {}; Given args len: {}",
692                func_ty.params.valtypes.len(),
693                param_types.len()
694            );
695            panic!("Invalid parameters for function");
696        }
697
698        // Prepare a new stack with the locals for the entry function
699        let mut stack = Stack::new();
700        let locals = Locals::new(params.into_iter(), func_inst.locals.iter().cloned());
701
702        // TODO handle this bad linear search that is unavoidable
703        let (func_idx, _) = self.modules[module_addr]
704            .func_addrs
705            .iter()
706            .enumerate()
707            .find(|&(_idx, addr)| *addr == func_addr)
708            .ok_or(RuntimeError::FunctionNotFound)?;
709
710        stack.push_stackframe(
711            module_addr,
712            func_idx,
713            &func_ty,
714            locals,
715            usize::MAX,
716            usize::MAX,
717        );
718
719        let mut currrent_module_idx = module_addr;
720        // Run the interpreter
721        run(
722            // &mut self.modules,
723            &mut currrent_module_idx,
724            // self.lut.as_ref().ok_or(RuntimeError::UnmetImport)?,
725            &mut stack,
726            EmptyHookSet,
727            self,
728        )?;
729
730        // Pop return values from stack
731        let return_values = func_ty
732            .returns
733            .valtypes
734            .iter()
735            .rev()
736            .map(|ty| stack.pop_value(*ty))
737            .collect::<Vec<Value>>();
738
739        // Values are reversed because they were popped from stack one-by-one. Now reverse them back
740        let reversed_values = return_values.into_iter().rev();
741        let ret = reversed_values.collect();
742        debug!("Successfully invoked function");
743        Ok(ret)
744    }

747    pub fn get_module_idx_from_name(&self, module_name: &str) -> Result<usize, RuntimeError> {
748        self.module_names
749            .get(module_name)
750            .copied()
751            .ok_or(RuntimeError::ModuleNotFound)
752    }

756        &self,

761            .get(module_addr)?

957/// common convention functions defined for lists of ExternVals, ExternTypes, Exports
958/// <https://webassembly.github.io/spec/core/exec/runtime.html#conventions>
959/// <https://webassembly.github.io/spec/core/syntax/types.html#id3>
960/// <https://webassembly.github.io/spec/core/syntax/modules.html?highlight=convention#id1>
961// TODO implement this trait for ExternType lists Export lists
962pub trait ExternFilterable<T> {
963    fn funcs(self) -> impl Iterator<Item = T>;
964    fn tables(self) -> impl Iterator<Item = T>;
965    fn mems(self) -> impl Iterator<Item = T>;
966    fn globals(self) -> impl Iterator<Item = T>;
967}
968
969impl<'a, I> ExternFilterable<usize> for I
970where
971    I: Iterator<Item = &'a ExternVal>,
972{
973    fn funcs(self) -> impl Iterator<Item = usize> {
974        self.filter_map(|extern_val| {
975            if let ExternVal::Func(func_addr) = extern_val {
976                Some(*func_addr)
977            } else {
978                None
979            }
980        })
981    }
982
983    fn tables(self) -> impl Iterator<Item = usize> {
984        self.filter_map(|extern_val| {
985            if let ExternVal::Table(table_addr) = extern_val {
986                Some(*table_addr)
987            } else {
988                None
989            }
990        })
991    }
992
993    fn mems(self) -> impl Iterator<Item = usize> {
994        self.filter_map(|extern_val| {
995            if let ExternVal::Mem(mem_addr) = extern_val {
996                Some(*mem_addr)
997            } else {
998                None
999            }
1000        })
1001    }
1002
1003    fn globals(self) -> impl Iterator<Item = usize> {
1004        self.filter_map(|extern_val| {
1005            if let ExternVal::Global(global_addr) = extern_val {
1006                Some(*global_addr)
1007            } else {
1008                None
1009            }
1010        })
1011    }
1012}
