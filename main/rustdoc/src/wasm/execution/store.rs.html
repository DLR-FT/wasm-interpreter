37/// The store represents all global state that can be manipulated by WebAssembly programs. It
38/// consists of the runtime representation of all instances of functions, tables, memories, and
39/// globals, element segments, and data segments that have been allocated during the life time of
40/// the abstract machine.
41/// <https://webassembly.github.io/spec/core/exec/runtime.html#store>
42pub struct Store<'b, T: Config> {
43    pub functions: Vec<FuncInst<T>>,
44    pub memories: Vec<MemInst>,
45    pub globals: Vec<GlobalInst>,
46    pub data: Vec<DataInst>,
47    pub tables: Vec<TableInst>,
48    pub elements: Vec<ElemInst>,
49    pub modules: Vec<ModuleInst<'b>>,
50
51    // fields outside of the spec but are convenient are below
52
53    // all visible exports and entities added by hand or module instantiation by the interpreter
54    // currently, all of the exports of an instantiated module is made visible (this is outside of spec)
55    pub registry: Registry,
56    pub user_data: T,
57
58    // data structure holding all resumable objects that belong to this store
59    pub(crate) dormitory: Dormitory,
60}

79    /// instantiates a validated module with `validation_info` as validation evidence with name `name`
80    /// with the steps in <https://webassembly.github.io/spec/core/exec/modules.html#instantiation>
81    /// this method roughly matches the suggested embedder function`module_instantiate`
82    /// <https://webassembly.github.io/spec/core/appendix/embedding.html#modules>
83    /// except external values for module instantiation are retrieved from `self`.
84    pub fn add_module(
85        &mut self,
86        name: &str,
87        validation_info: &ValidationInfo<'b>,
88        maybe_fuel: Option<u32>,
89    ) -> Result<(), RuntimeError> {
90        // instantiation step -1: collect extern_vals, this section basically acts as a linker between modules
91        // best attempt at trying to match the spec implementation in terms of errors
92        debug!("adding module with name {:?}", name);
93        let mut extern_vals = Vec::new();
94
95        for Import {
96            module_name: exporting_module_name,
97            name: import_name,
98            desc: import_desc,
99        } in &validation_info.imports
100        {
101            trace!(
102                "trying to import from exporting module instance named {:?}, the entity with name {:?} with desc: {:?}",
103                exporting_module_name,
104                import_name,
105                import_desc
106            );
107            let import_extern_type = import_desc.extern_type(validation_info);
108            let export_extern_val_candidate = *self.registry.lookup(
109                exporting_module_name.clone().into(),
110                import_name.clone().into(),
111            )?;
112            trace!("export candidate found: {:?}", export_extern_val_candidate);
113            if !export_extern_val_candidate
114                .extern_type(self)
115                .is_subtype_of(&import_extern_type)
116            {
117                return Err(RuntimeError::InvalidImportType);
118            }
119            trace!("import and export matches. Adding to externvals");
120            extern_vals.push(export_extern_val_candidate)
121        }
122
123        // instantiation: step 5
124        // module_inst_init is unfortunately circularly defined from parts of module_inst that would be defined in step 11, which uses module_inst_init again implicitly.
125        // therefore I am mimicking the reference interpreter code here, I will allocate functions in the store in this step instead of step 11.
126        // https://github.com/WebAssembly/spec/blob/8d6792e3d6709e8d3e90828f9c8468253287f7ed/interpreter/exec/eval.ml#L789
127        let mut module_inst = ModuleInst {
128            types: validation_info.types.clone(),
129            func_addrs: extern_vals.iter().funcs().collect(),
130            table_addrs: Vec::new(),
131            mem_addrs: Vec::new(),
132            global_addrs: extern_vals.iter().globals().collect(),
133            elem_addrs: Vec::new(),
134            data_addrs: Vec::new(),
135            exports: BTreeMap::new(),
136            wasm_bytecode: validation_info.wasm,
137            sidetable: validation_info.sidetable.clone(),
138        };
139
140        // TODO rewrite this part
141        // <https://webassembly.github.io/spec/core/exec/modules.html#functions>
142        let func_addrs: Vec<usize> = validation_info
143            .functions
144            .iter()
145            .zip(validation_info.func_blocks_stps.iter())
146            .map(|(ty_idx, (span, stp))| {
147                self.alloc_func((*ty_idx, (*span, *stp)), &module_inst, self.modules.len())
148            })
149            .collect();
150
151        module_inst.func_addrs.extend(func_addrs);
152
153        // instantiation: this roughly matches step 6,7,8
154        // validation guarantees these will evaluate without errors.
155        let maybe_global_init_vals: Result<Vec<Value>, _> = validation_info
156            .globals
157            .iter()
158            .map(|global| {
159                run_const_span(validation_info.wasm, &global.init_expr, &module_inst, self)
160                    .transpose()
161                    .unwrap_validated()
162            })
163            .collect();
164        let global_init_vals = maybe_global_init_vals?;
165
166        // instantiation: this roughly matches step 9,10
167
168        let mut element_init_ref_lists: Vec<Vec<Ref>> =
169            Vec::with_capacity(validation_info.elements.len());
170
171        for elem in &validation_info.elements {
172            let mut new_list = Vec::new();
173            match &elem.init {
174                // shortcut of evaluation of "ref.func <func_idx>; end;"
175                // validation guarantees corresponding func_idx's existence
176                ElemItems::RefFuncs(ref_funcs) => {
177                    for func_idx in ref_funcs {
178                        let func_addr = *module_inst
179                            .func_addrs
180                            .get(*func_idx as usize)
181                            .unwrap_validated();
182
183                        new_list.push(Ref::Func(FuncAddr(func_addr)));
184                    }
185                }
186                ElemItems::Exprs(_, exprs) => {
187                    for expr in exprs {
188                        new_list.push(
189                            run_const_span(validation_info.wasm, expr, &module_inst, self)?
190                                .unwrap_validated() // there is a return value
191                                .try_into()
192                                .unwrap_validated(), // return value has the correct type
193                        )
194                    }
195                }
196            }
197            element_init_ref_lists.push(new_list);
198        }
199
200        // instantiation: step 11 - module allocation (except function allocation - which was made in step 5)
201        // https://webassembly.github.io/spec/core/exec/modules.html#alloc-module
202
203        // allocation: begin
204
205        // allocation: step 1
206        let module = validation_info;
207
208        let extern_vals = extern_vals;
209        let vals = global_init_vals;
210        let ref_lists = element_init_ref_lists;
211
212        // allocation: skip step 2 as it was done in instantiation step 5
213
214        // allocation: step 3-13
215        let table_addrs: Vec<usize> = module
216            .tables
217            .iter()
218            .map(|table_type| self.alloc_table(*table_type, Ref::Null(table_type.et)))
219            .collect();
220        let mem_addrs: Vec<usize> = module
221            .memories
222            .iter()
223            .map(|mem_type| self.alloc_mem(*mem_type))
224            .collect();
225        let global_addrs: Vec<usize> = module
226            .globals
227            .iter()
228            .zip(vals)
229            .map(
230                |(
231                    Global {
232                        ty: global_type, ..
233                    },
234                    val,
235                )| self.alloc_global(*global_type, val),
236            )
237            .collect();
238        let elem_addrs = module
239            .elements
240            .iter()
241            .zip(ref_lists)
242            .map(|(elem, refs)| self.alloc_elem(elem.ty(), refs))
243            .collect();
244        let data_addrs = module
245            .data
246            .iter()
247            .map(|DataSegment { init: bytes, .. }| self.alloc_data(bytes))
248            .collect();
249
250        // allocation: skip step 14 as it was done in instantiation step 5
251
252        // allocation: step 15,16
253        let mut table_addrs_mod: Vec<usize> = extern_vals.iter().tables().collect();
254        table_addrs_mod.extend(table_addrs);
255
256        let mut mem_addrs_mod: Vec<usize> = extern_vals.iter().mems().collect();
257        mem_addrs_mod.extend(mem_addrs);
258
259        // skipping step 17 partially as it was partially done in instantiation step
260        module_inst.global_addrs.extend(global_addrs);
261
262        // allocation: step 18,19
263        let export_insts: BTreeMap<String, ExternVal> = module
264            .exports
265            .iter()
266            .map(|Export { name, desc }| {
267                let value = match desc {
268                    ExportDesc::FuncIdx(func_idx) => {
269                        ExternVal::Func(module_inst.func_addrs[*func_idx])
270                    }
271                    ExportDesc::TableIdx(table_idx) => {
272                        ExternVal::Table(table_addrs_mod[*table_idx])
273                    }
274                    ExportDesc::MemIdx(mem_idx) => ExternVal::Mem(mem_addrs_mod[*mem_idx]),
275                    ExportDesc::GlobalIdx(global_idx) => {
276                        ExternVal::Global(module_inst.global_addrs[*global_idx])
277                    }
278                };
279                (String::from(name), value)
280            })
281            .collect();
282
283        // allocation: step 20,21 initialize module (except functions and globals due to instantiation step 5, allocation step 14,17)
284        module_inst.table_addrs = table_addrs_mod;
285        module_inst.mem_addrs = mem_addrs_mod;
286        module_inst.elem_addrs = elem_addrs;
287        module_inst.data_addrs = data_addrs;
288        module_inst.exports = export_insts;
289
290        // allocation: end
291
292        // register module exports, this is outside of the spec
293        self.registry
294            .register_module(name.to_owned().into(), &module_inst)?;
295
296        // instantiation step 11 end: module_inst properly allocated after this point.
297        // TODO: it is too hard with our codebase to do the following steps without adding the module to the store
298        let current_module_idx = self.modules.len();
299        self.modules.push(module_inst);
300
301        // instantiation: step 12-15
302        // TODO have to stray away from the spec a bit since our codebase does not lend itself well to freely executing instructions by themselves
303        for (
304            i,
305            ElemType {
306                init: elem_items,
307                mode,
308            },
309        ) in validation_info.elements.iter().enumerate()
310        {
311            match mode {
312                ElemMode::Active(ActiveElem {
313                    table_idx: table_idx_i,
314                    init_expr: einstr_i,
315                }) => {
316                    let n = elem_items.len() as i32;
317                    // equivalent to init.len() in spec
318                    // instantiation step 14:
319                    // TODO (for now, we are doing hopefully what is equivalent to it)
320                    // execute:
321                    //   einstr_i
322                    //   i32.const 0
323                    //   i32.const n
324                    //   table.init table_idx_i i
325                    //   elem.drop i
326                    let d: i32 = run_const_span(
327                        validation_info.wasm,
328                        einstr_i,
329                        &self.modules[current_module_idx],
330                        self,
331                    )?
332                    .unwrap_validated() // there is a return value
333                    .try_into()
334                    .unwrap_validated(); // return value has correct type
335
336                    let s = 0;
337                    table_init(
338                        &self.modules,
339                        &mut self.tables,
340                        &self.elements,
341                        current_module_idx,
342                        i,
343                        *table_idx_i as usize,
344                        n,
345                        s,
346                        d,
347                    )?;
348                    elem_drop(&self.modules, &mut self.elements, current_module_idx, i)?;
349                }
350                ElemMode::Declarative => {
351                    // instantiation step 15:
352                    // TODO (for now, we are doing hopefully what is equivalent to it)
353                    // execute:
354                    //   elem.drop i
355                    elem_drop(&self.modules, &mut self.elements, current_module_idx, i)?;
356                }
357                ElemMode::Passive => (),
358            }
359        }
360
361        // instantiation: step 16
362        // TODO have to stray away from the spec a bit since our codebase does not lend itself well to freely executing instructions by themselves
363        for (i, DataSegment { init, mode }) in validation_info.data.iter().enumerate() {
364            match mode {
365                crate::core::reader::types::data::DataMode::Active(DataModeActive {
366                    memory_idx,
367                    offset: dinstr_i,
368                }) => {
369                    let n = init.len() as i32;
370                    // assert: mem_idx is 0
371                    if *memory_idx != 0 {
372                        // TODO fix error
373                        return Err(RuntimeError::MoreThanOneMemory);
374                    }
375
376                    // TODO (for now, we are doing hopefully what is equivalent to it)
377                    // execute:
378                    //   dinstr_i
379                    //   i32.const 0
380                    //   i32.const n
381                    //   memory.init i
382                    //   data.drop i
383                    let d: i32 = run_const_span(
384                        validation_info.wasm,
385                        dinstr_i,
386                        &self.modules[current_module_idx],
387                        self,
388                    )?
389                    .unwrap_validated() // there is a return value
390                    .try_into()
391                    .unwrap_validated(); // return value has the correct type
392
393                    let s = 0;
394                    memory_init(
395                        &self.modules,
396                        &mut self.memories,
397                        &self.data,
398                        current_module_idx,
399                        i,
400                        0,
401                        n,
402                        s,
403                        d,
404                    )?;
405                    data_drop(&self.modules, &mut self.data, current_module_idx, i)?;
406                }
407                crate::core::reader::types::data::DataMode::Passive => (),
408            }
409        }
410
411        // instantiation: step 17
412        if let Some(func_idx) = validation_info.start {
413            // TODO (for now, we are doing hopefully what is equivalent to it)
414            // execute
415            //   call func_ifx
416            let func_addr = self.modules[current_module_idx].func_addrs[func_idx];
417            self.invoke(func_addr, Vec::new(), maybe_fuel)?;
418        };
419
420        Ok(())

426    fn alloc_func(
427        &mut self,
428        func: (TypeIdx, (Span, usize)),
429        module_inst: &ModuleInst,
430        module_addr: usize,
431    ) -> usize {
432        let (ty, (span, stp)) = func;
433
434        // TODO rewrite this huge chunk of parsing after generic way to re-parse(?) structs lands
435        let mut wasm_reader = WasmReader::new(module_inst.wasm_bytecode);
436        wasm_reader.move_start_to(span).unwrap_validated();
437
438        let (locals, bytes_read) = wasm_reader
439            .measure_num_read_bytes(crate::code::read_declared_locals)
440            .unwrap_validated();
441
442        let code_expr = wasm_reader
443            .make_span(span.len() - bytes_read)
444            .unwrap_validated();
445
446        // core of the method below
447
448        // validation guarantees func_ty_idx exists within module_inst.types
449        // TODO fix clone
450        let func_inst = FuncInst::WasmFunc(WasmFuncInst {
451            function_type: module_inst.types[ty].clone(),
452            ty,
453            locals,
454            code_expr,
455            stp,
456            module_addr,
457        });
458
459        let addr = self.functions.len();
460        self.functions.push(func_inst);
461        addr
462    }

566    /// Sets a new value of some global instance by its addr.
567    ///
568    /// # Errors
569    /// - [`RuntimeError::WriteOnImmutableGlobal`]
570    /// - [`RuntimeError::GlobalTypeMismatch`]
571    ///
572    /// See: WebAssembly Specification 2.0 - 7.1.10 - global_write
573    pub fn global_write(&mut self, global_addr: usize, val: Value) -> Result<(), RuntimeError> {
574        // 1. Let `gi` be the global instance `store.globals[globaladdr]`.
575        let gi = self
576            .globals
577            .get_mut(global_addr)
578            .expect("global addrs to always be valid if the correct store is used");
579
580        // 2. Let `mut t` be the structure of the global type `gi.type`.
581        let r#mut = gi.ty.is_mut;
582        let t = gi.ty.ty;
583
584        // 3. If `mut` is not `var`, then return error.
585        if !r#mut {
586            return Err(RuntimeError::WriteOnImmutableGlobal);
587        }
588
589        // Check invariant:
590        //   It is an invariant of the semantics that the value has a type equal to the value type of `globaltype`.
591        // See: WebAssembly Specification 2.0 - 4.2.9
592        if t != val.to_ty() {
593            return Err(RuntimeError::GlobalTypeMismatch);
594        }
595
596        // 4. Replace `gi.value` with the value `val`.
597        gi.value = val;
598
599        // 5. Return the updated store.
600        // This is a noop for us, as our store `self` is mutable.
601
602        Ok(())
603    }

605    pub fn create_resumable(
606        &self,
607        func_addr: usize,
608        params: Vec<Value>,
609        maybe_fuel: Option<u32>,
610    ) -> Result<ResumableRef, RuntimeError> {
611        let func_inst = self
612            .functions
613            .get(func_addr)
614            .ok_or(RuntimeError::FunctionNotFound)?;
615
616        let func_ty = func_inst.ty();
617
618        // Verify that the given parameters match the function parameters
619        let param_types = params.iter().map(|v| v.to_ty()).collect::<Vec<_>>();
620
621        if func_ty.params.valtypes != param_types {
622            trace!(
623                "Func param types len: {}; Given args len: {}",
624                func_ty.params.valtypes.len(),
625                param_types.len()
626            );
627            panic!("Invalid parameters for function");
628        }
629
630        Ok(ResumableRef::Fresh(FreshResumableRef {
631            func_addr,
632            params,
633            maybe_fuel,
634        }))
635    }

637    pub fn resume(&mut self, mut resumable_ref: ResumableRef) -> Result<RunState, RuntimeError> {
638        match resumable_ref {
639            ResumableRef::Fresh(FreshResumableRef {
640                func_addr,
641                params,
642                maybe_fuel,
643            }) => {
644                let func_inst = self
645                    .functions
646                    .get(func_addr)
647                    .expect("func addrs to always be valid if the correct store is used");
648
649                match func_inst {
650                    FuncInst::HostFunc(host_func_inst) => {
651                        let returns = (host_func_inst.hostcode)(&mut self.user_data, params);
652                        debug!("Successfully invoked function");
653
654                        let returns = returns.map_err(|HaltExecutionError| {
655                            RuntimeError::HostFunctionHaltedExecution
656                        })?;
657
658                        // Verify that the return parameters match the host function parameters
659                        // since we have no validation guarantees for host functions
660
661                        let return_types = returns.iter().map(|v| v.to_ty()).collect::<Vec<_>>();
662                        if host_func_inst.function_type.returns.valtypes != return_types {
663                            trace!(
664                                "Func return types len: {}; returned args len: {}",
665                                host_func_inst.function_type.returns.valtypes.len(),
666                                return_types.len()
667                            );
668                            return Err(RuntimeError::HostFunctionSignatureMismatch);
669                        }
670
671                        Ok(RunState::Finished(returns))
672                    }
673                    FuncInst::WasmFunc(wasm_func_inst) => {
674                        // Prepare a new stack with the locals for the entry function
675                        let mut stack = Stack::new_with_values(params);
676
677                        stack.push_call_frame::<T>(
678                            usize::MAX,
679                            &wasm_func_inst.function_type,
680                            &wasm_func_inst.locals,
681                            usize::MAX,
682                            usize::MAX,
683                        )?;
684
685                        let mut resumable = Resumable {
686                            current_func_addr: func_addr,
687                            stack,
688                            pc: wasm_func_inst.code_expr.from,
689                            stp: wasm_func_inst.stp,
690                            maybe_fuel,
691                        };
692
693                        // Run the interpreter
694                        let result = interpreter_loop::run(&mut resumable, self)?;
695
696                        match result {
697                            None => {
698                                debug!("Successfully invoked function");
699                                Ok(RunState::Finished(resumable.stack.into_values()))
700                            }
701                            Some(required_fuel) => {
702                                debug!("Successfully invoked function, but ran out of fuel");
703                                Ok(RunState::Resumable {
704                                    resumable_ref: ResumableRef::Invoked(
705                                        self.dormitory.insert(resumable),
706                                    ),
707                                    required_fuel,
708                                })
709                            }
710                        }
711                    }
712                }
713            }
714            ResumableRef::Invoked(InvokedResumableRef {
715                dormitory: ref mut dormitory_weak,
716                ref key,
717            }) => {
718                // Resuming requires `self`'s dormitory to still be alive
719                let Some(dormitory) = dormitory_weak.upgrade() else {
720                    return Err(RuntimeError::ResumableNotFound);
721                };
722
723                // Check the given `RuntimeInstance` is the same one used to create `self`
724                if !Arc::ptr_eq(&dormitory, &self.dormitory.0) {
725                    return Err(RuntimeError::ResumableNotFound);
726                }
727
728                // Obtain a write lock to the `Dormitory`
729                let mut dormitory = dormitory.write();
730
731                // TODO We might want to remove the `Resumable` here already and later reinsert it.
732                // This would prevent holding the lock across the interpreter loop.
733                let resumable = dormitory
734                    .get_mut(key)
735                    .expect("the key to always be valid as self was not dropped yet");
736
737                // Resume execution
738                let result = interpreter_loop::run(resumable, self)?;
739
740                match result {
741                    None => {
742                        let resumable = dormitory.remove(key)
743                            .expect("that the resumable could not have been removed already, because then this self could not exist");
744
745                        // Take the `Weak` pointing to the dormitory out of `self` and replace it with a default `Weak`.
746                        // This causes the `Drop` impl of `self` to directly quit preventing it from unnecessarily locking the dormitory.
747                        let _dormitory = mem::take(dormitory_weak);
748
749                        Ok(RunState::Finished(resumable.stack.into_values()))
750                    }
751                    Some(required_fuel) => Ok(RunState::Resumable {
752                        resumable_ref,
753                        required_fuel,
754                    }),
755                }
756            }
757        }
758    }
759
760    pub fn access_fuel_mut<R>(
761        &mut self,
762        resumable_ref: &mut ResumableRef,
763        f: impl FnOnce(&mut Option<u32>) -> R,
764    ) -> Result<R, RuntimeError> {
765        match resumable_ref {
766            ResumableRef::Fresh(FreshResumableRef { maybe_fuel, .. }) => Ok(f(maybe_fuel)),
767            ResumableRef::Invoked(resumable_ref) => {
768                // Resuming requires `self`'s dormitory to still be alive
769                let Some(dormitory) = resumable_ref.dormitory.upgrade() else {
770                    return Err(RuntimeError::ResumableNotFound);
771                };
772
773                // Check the given `RuntimeInstance` is the same one used to create `self`
774                if !Arc::ptr_eq(&dormitory, &self.dormitory.0) {
775                    return Err(RuntimeError::ResumableNotFound);
776                }
777
778                let mut dormitory = dormitory.write();
779
780                let resumable = dormitory
781                    .get_mut(&resumable_ref.key)
782                    .expect("the key to always be valid as self was not dropped yet");
783
784                Ok(f(&mut resumable.maybe_fuel))
785            }
786        }
787    }
788
789    pub fn invoke(
790        &mut self,
791        func_addr: usize,
792        params: Vec<Value>,
793        maybe_fuel: Option<u32>,
794    ) -> Result<RunState, RuntimeError> {
795        self.resume(self.create_resumable(func_addr, params, maybe_fuel)?)
796    }

882        let len = n
883            .checked_add(self.elem.len() as u32)
884            .ok_or(TrapError::TableOrElementAccessOutOfBounds)?;
885
886        // roughly matches step 4,5,6
887        // checks limits_prime.valid() for limits_prime := { min: len, max: self.ty.lim.max }
888        // https://webassembly.github.io/spec/core/valid/types.html#limits
889        if self.ty.lim.max.map(|max| len > max).unwrap_or(false) {
890            return Err(TrapError::TableOrElementAccessOutOfBounds.into());
891        }
892        let limits_prime = Limits {
893            min: len,
894            max: self.ty.lim.max,
895        };
896
897        self.elem.extend(vec![reff; n as usize]);
898
899        self.ty.lim = limits_prime;
900        Ok(())

1011                store
1012                    .memories
1013                    .get(*mem_addr)
1014                    .expect("the correct store to be used")
1015                    .ty,

1018                store
1019                    .globals

1041where
1042    I: Iterator<Item = &'a ExternVal>,
1043{

1045        self.filter_map(|extern_val| {

1048            } else {
1049                None
1050            }
1051        })

1057                Some(*table_addr)
1058            } else {
1059                None
1060            }
1061        })
1062    }
1063
1064    fn mems(self) -> impl Iterator<Item = usize> {
1065        self.filter_map(|extern_val| {
1066            if let ExternVal::Mem(mem_addr) = extern_val {
1067                Some(*mem_addr)
1068            } else {
1069                None
1070            }
1071        })
1072    }
1073
1074    fn globals(self) -> impl Iterator<Item = usize> {
1075        self.filter_map(|extern_val| {
1076            if let ExternVal::Global(global_addr) = extern_val {
1077                Some(*global_addr)
1078            } else {
1079                None
1080            }
1081        })
1082    }
