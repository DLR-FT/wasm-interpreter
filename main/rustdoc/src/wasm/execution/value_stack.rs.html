15/// The stack at runtime containing
16/// 1. Values
17/// 2. Labels
18/// 3. Activations
19///
20/// See <https://webassembly.github.io/spec/core/exec/runtime.html#stack>
21#[derive(Default, Debug)]
22pub(crate) struct Stack {
23    /// WASM values on the stack, i.e. the actual data that instructions operate on
24    values: Vec<Value>,
25
26    /// Call frames
27    ///
28    /// Each time a function is called, a new frame is pushed, whenever a function returns, a frame is popped
29    frames: Vec<CallFrame>,
30}

48    pub fn drop_value(&mut self) {
49        // If there is at least one call frame, we shall not pop values past the current
50        // call frame. However, there is one legitimate reason to pop when there is **no** current
51        // call frame: after the outermost function returns, to extract the final return values of
52        // this interpreter invocation.
53        debug_assert!(
54            if !self.frames.is_empty() {
55                self.values.len() > self.current_call_frame().value_stack_base_idx
56            } else {
57                true
58            },
59            "can not pop values past the current call frame"
60        );
61
62        self.values.pop().unwrap_validated();
63    }

65    /// Pop a reference of unknown type from the value stack
66    pub fn pop_unknown_ref(&mut self) -> Ref {
67        // If there is at least one call frame, we shall not pop values past the current
68        // call frame. However, there is one legitimate reason to pop when there is **no** current
69        // call frame: after the outermost function returns, to extract the final return values of
70        // this interpreter invocation.
71        debug_assert!(
72            if !self.frames.is_empty() {
73                self.values.len() > self.current_call_frame().value_stack_base_idx
74            } else {
75                true
76            },
77            "can not pop values past the current call frame"
78        );
79
80        let popped = self.values.pop().unwrap_validated();
81        match popped.to_ty() {
82            ValType::RefType(_) => match popped {
83                Value::Ref(rref) => rref,
84                _ => unreachable!(),
85            },
86            _ => unreachable_validated!(),
87        }
88    }

99            } else {

113    /// Copy a value of the given [ValType] from the value stack without removing it
114    pub fn peek_value(&self, ty: ValType) -> Value {
115        let value = self.values.last().unwrap_validated();
116        if value.to_ty() == ty {
117            *value
118        } else {
119            unreachable_validated!()
120        }
121    }

123    /// Returns a cloned copy of the top value on the stack, or `None` if the stack is empty
124    pub fn peek_unknown_value(&self) -> Option<Value> {
125        self.values.last().copied()
126    }

128    /// Push a value to the value stack
129    pub fn push_value(&mut self, value: Value) -> Result<(), RuntimeError> {
130        // check for value stack exhaustion
131        if self.values.len() > MAX_VALUE_STACK_SIZE {
132            return Err(RuntimeError::StackExhaustion);
133        }
134
135        // push the value
136        self.values.push(value);
137
138        Ok(())
139    }

141    /// Copy a local variable to the top of the value stack
142    pub fn get_local(&mut self, idx: LocalIdx) -> Result<(), RuntimeError> {
143        let call_frame_base_idx = self.current_call_frame().call_frame_base_idx;
144        let local_value = self
145            .values
146            .get(call_frame_base_idx + idx)
147            .unwrap_validated();
148        self.push_value(*local_value)
149    }

151    /// Pop value from the top of the value stack, writing it to the given local
152    pub fn set_local(&mut self, idx: LocalIdx) {
153        debug_assert!(
154            self.values.len() > self.current_call_frame().value_stack_base_idx,
155            "can not pop values past the current call frame"
156        );
157
158        let call_frame_base_idx = self.current_call_frame().call_frame_base_idx;
159        let stack_value = self.pop_value();
160
161        trace!("Instruction: local.set [{stack_value:?}] -> []");
162
163        *self
164            .values
165            .get_mut(call_frame_base_idx + idx)
166            .unwrap_validated() = stack_value;
167    }

169    /// Copy value from top of the value stack to the given local
170    pub fn tee_local(&mut self, idx: LocalIdx) {
171        let call_frame_base_idx = self.current_call_frame().call_frame_base_idx;
172
173        let local_ty = self
174            .values
175            .get(call_frame_base_idx + idx)
176            .unwrap_validated()
177            .to_ty();
178        let stack_value = self.peek_value(local_ty);
179
180        trace!("Instruction: local.tee [{stack_value:?}] -> []");
181
182        *self
183            .values
184            .get_mut(call_frame_base_idx + idx)
185            .unwrap_validated() = stack_value;
186    }

198    /// Pop a [`CallFrame`] from the call stack, returning the caller function store address, return address, and the return stp
199    pub fn pop_call_frame(&mut self) -> (usize, usize, usize) {
200        let CallFrame {
201            return_func_addr,
202            return_addr,
203            call_frame_base_idx,
204            return_value_count,
205            return_stp,
206            ..
207        } = self.frames.pop().unwrap_validated();
208
209        let remove_count = self.values.len() - call_frame_base_idx - return_value_count;
210
211        self.remove_inbetween(remove_count, return_value_count);
212
213        debug_assert_eq!(
214            self.values.len(),
215            call_frame_base_idx + return_value_count,
216            "after a function call finished, the stack must have exactly as many values as it had before calling the function plus the number of function return values"
217        );
218
219        (return_func_addr, return_addr, return_stp)
220    }
221
222    /// Push a call frame to the call stack
223    ///
224    /// Takes the current [`Self::values`]'s length as [`CallFrame::value_stack_base_idx`].
225    pub fn push_call_frame(
226        &mut self,
227        return_func_addr: usize,
228        func_ty: &FuncType,
229        remaining_locals: &[ValType],
230        return_addr: usize,
231        return_stp: usize,
232    ) -> Result<(), RuntimeError> {
233        // check for call stack exhaustion
234        if self.call_frame_count() > MAX_CALL_STACK_SIZE {
235            return Err(RuntimeError::StackExhaustion);
236        }
237
238        debug_assert!(
239            self.values.len() >= func_ty.params.valtypes.len(),
240            "when pushing a new call frame, at least as many values need to be on the stack as required by the new call frames's function"
241        );
242
243        // the topmost `param_count` values are transferred into/consumed by this new call frame
244        let param_count = func_ty.params.valtypes.len();
245        let call_frame_base_idx = self.values.len() - param_count;
246
247        // after the params, put the additional locals
248        for local in remaining_locals {
249            self.values.push(Value::default_from_ty(*local));
250        }
251
252        // now that the locals are all populated, the actual stack section of this call frame begins
253        let value_stack_base_idx = self.values.len();
254
255        self.frames.push(CallFrame {
256            return_func_addr,
257            return_addr,
258            value_stack_base_idx,
259            call_frame_base_idx,
260            return_value_count: func_ty.returns.valtypes.len(),
261            return_stp,
262        });
263
264        Ok(())
265    }
266
267    /// Returns how many call frames are on the stack, in total.
268    pub fn call_frame_count(&self) -> usize {
269        self.frames.len()
270    }
271
272    /// Pop `n` elements from the value stack's tail as an iterator, with the first element being
273    /// closest to the **bottom** of the value stack
274    ///
275    /// Note that this is providing the values in reverse order compared to popping `n` values
276    /// (which would yield the element closest to the **top** of the value stack first).
277    pub fn pop_tail_iter(&mut self, n: usize) -> Drain<'_, Value> {
278        let start = self.values.len() - n;
279        self.values.drain(start..)
280    }
281
282    /// Remove `remove_count` values from the stack, keeping the topmost `keep_count` values
283    ///
284    /// From the stack, remove `remove_count` elements, by sliding down the `keep_count` topmost
285    /// values `remove_count` positions.
286    ///
287    /// **Effects**
288    ///
289    /// - after the operation, [`Stack`] will contain `remove_count` fewer elements
290    /// - `keep_count` topmost elements will be identical before and after the operation
291    /// - all elements below the `remove_count + keep_count` topmost stack entry remain
292    pub fn remove_inbetween(&mut self, remove_count: usize, keep_count: usize) {
293        let len = self.values.len();
294        self.values
295            .copy_within(len - keep_count.., len - keep_count - remove_count);
296        self.values.truncate(len - remove_count);
297    }

300/// The [WASM spec](https://webassembly.github.io/spec/core/exec/runtime.html#stack) calls this `Activations`, however it refers to the call frames of functions.
301#[derive(Debug)]
302pub(crate) struct CallFrame {
303    /// Store address of the function that called this [`CallFrame`]'s function
304    pub return_func_addr: usize,
305
306    /// Value that the PC has to be set to when this function returns
307    pub return_addr: usize,
308
309    /// The index to the lowermost value in [`Stack::values`] belonging to this [`CallFrame`]'s
310    /// stack
311    ///
312    /// Values below this may still belong to this [`CallFrame`], but they are locals. Consequently,
313    /// this is the lowest index down to which the stack may be popped in this [`CallFrame`].
314    /// However, clearing up this [`CallFrame`] may require further popping, down to (and
315    /// including!) the index stored in [`Self::call_frame_base_idx`].
316    pub value_stack_base_idx: usize,
317
318    /// The index to the lowermost value on [`Stack::values`] that belongs to this [`CallFrame`]
319    ///
320    /// Clearing this [`CallFrame`] requires popping all elements on [`Stack::values`] down to (and
321    /// including!) this index.
322    pub call_frame_base_idx: usize,
323
324    /// Number of return values to retain on [`Stack::values`] when unwinding/popping a [`CallFrame`]
325    pub return_value_count: usize,
326
327    // Value that the stp has to be set to when this function returns
328    pub return_stp: usize,
329}
