32/// The default module name if a [RuntimeInstance] was created using [RuntimeInstance::new].

35pub struct RuntimeInstance<'b, T = (), H = EmptyHookSet>

37    H: HookSet + core::fmt::Debug,

104        &self,
105        module_addr: usize,
106        function_idx: usize,
107    ) -> Result<FunctionRef, RuntimeError> {
108        let module_inst = self
109            .store
110            .modules
111            .get(module_addr)
112            .ok_or(RuntimeError::ModuleNotFound)?;
113        let func_addr = *module_inst
114            .func_addrs
115            .get(function_idx)
116            .ok_or(RuntimeError::FunctionNotFound)?;
117
118        Ok(FunctionRef { func_addr })
119    }

121    /// Invokes a function with the given parameters of type `Param`, and return types of type `Returns`.
122    pub fn invoke_typed<Params: InteropValueList, Returns: InteropValueList>(
123        &mut self,
124        function_ref: &FunctionRef,
125        params: Params,
126        // store: &mut Store,
127    ) -> Result<Returns, RuntimeError> {
128        self.invoke(function_ref, params.into_values())
129            .map(|values| Returns::try_from_values(values.into_iter()).unwrap_validated())
130    }

132    /// Invokes a function with the given parameters. The return types depend on the function signature.
133    pub fn invoke(
134        &mut self,
135        function_ref: &FunctionRef,
136        params: Vec<Value>,
137    ) -> Result<Vec<Value>, RuntimeError> {
138        let FunctionRef { func_addr } = *function_ref;
139        self.store
140            .invoke(func_addr, params, None)
141            .map(|run_state| match run_state {
142                RunState::Finished(values) => values,
143                _ => unreachable!("non metered invoke call"),
144            })
145    }

166    /// calls its argument `f` with a mutable reference of the fuel of the respective [`ResumableRef`].
167    ///
168    /// Fuel is stored as an [`Option<u32>`], where `None` means that fuel is disabled and `Some(x)` means that `x` units of fuel is left.
169    /// A ubiquitious use of this method would be using `f` to read or mutate the current fuel amount of the respective [`ResumableRef`].
170    /// # Example
171    /// ```
172    /// use wasm::{resumable::RunState, validate, RuntimeInstance};
173    /// let wasm = [ 0x00, 0x61, 0x73, 0x6d, 0x01, 0x00, 0x00, 0x00,
174    ///             0x01, 0x04, 0x01, 0x60, 0x00, 0x00, 0x03, 0x02,
175    ///             0x01, 0x00, 0x07, 0x09, 0x01, 0x05, 0x6c, 0x6f,
176    ///             0x6f, 0x70, 0x73, 0x00, 0x00, 0x0a, 0x09, 0x01,
177    ///             0x07, 0x00, 0x03, 0x40, 0x0c, 0x00, 0x0b, 0x0b ];
178    /// // a simple module with a single function looping forever
179    /// let mut instance = RuntimeInstance::new_named((), "module", &validate(&wasm).unwrap()).unwrap();
180    /// let func_ref = instance.get_function_by_name("module", "loops").unwrap();
181    /// let mut resumable_ref = instance.create_resumable(&func_ref, vec![], 0).unwrap();
182    /// instance.access_fuel_mut(&mut resumable_ref, |x| { assert_eq!(*x, Some(0)); *x = None; }).unwrap();
183    /// ```
184    pub fn access_fuel_mut<R>(
185        &mut self,
186        resumable_ref: &mut ResumableRef,
187        f: impl FnOnce(&mut Option<u32>) -> R,
188    ) -> Result<R, RuntimeError> {
189        self.store.access_fuel_mut(resumable_ref, f)
190    }

192    /// Adds a host function under module namespace `module_name` with name `name`.
193    /// roughly similar to `func_alloc` in <https://webassembly.github.io/spec/core/appendix/embedding.html#functions>
194    /// except the host function is made visible to other modules through these names.
195    pub fn add_host_function_typed<Params: InteropValueList, Returns: InteropValueList>(
196        &mut self,
197        module_name: &str,
198        name: &str,
199        host_func: fn(&mut T, Vec<Value>) -> Vec<Value>,
200    ) -> Result<FunctionRef, RuntimeError> {
201        let host_func_ty = FuncType {
202            params: ResultType {
203                valtypes: Vec::from(Params::TYS),
204            },
205            returns: ResultType {
206                valtypes: Vec::from(Returns::TYS),
207            },
208        };
209        self.add_host_function(module_name, name, host_func_ty, host_func)
210    }
211
212    pub fn add_host_function(
213        &mut self,
214        module_name: &str,
215        name: &str,
216        host_func_ty: FuncType,
217        host_func: fn(&mut T, Vec<Value>) -> Vec<Value>,
218    ) -> Result<FunctionRef, RuntimeError> {
219        let func_addr = self.store.alloc_host_func(host_func_ty, host_func);
220        self.store.registry.register(
221            module_name.to_owned().into(),
222            name.to_owned().into(),
223            store::ExternVal::Func(func_addr),
224        )?;
225        Ok(FunctionRef { func_addr })
226    }

237/// Helper function to quickly construct host functions without worrying about wasm to Rust
238/// type conversion. For user data, simply move the mutable reference into the passed closure.
239/// # Example
240/// ```
241/// use wasm::{validate, RuntimeInstance, host_function_wrapper, Value};
242/// fn my_wrapped_host_func(user_data: &mut (), params: Vec<Value>) -> Vec<Value> {
243///     host_function_wrapper(params, |(x, y): (u32, i32)| -> u32 {
244///         let _user_data = user_data;
245///         x + (y as u32)
246///  })
247/// }
248/// fn main() {
249///     let mut instance = RuntimeInstance::new(());
250///     let foo_bar = instance.add_host_function_typed::<(u32,i32),u32>("foo", "bar", my_wrapped_host_func).unwrap();
251/// }
252/// ```
253pub fn host_function_wrapper<Params: InteropValueList, Results: InteropValueList>(
254    params: Vec<Value>,
255    f: impl FnOnce(Params) -> Results,
256) -> Vec<Value> {
257    let params =
258        Params::try_from_values(params.into_iter()).expect("Params match the actual parameters");
259    let results = f(params);
260    results.into_values()
261}
