8/// Read and validate constant expressions.
9///
10/// This function is used to validate that a constant expression produces the expected result. The main use case for
11/// this is to validate that an initialization expression for a global returns the correct value.
12///
13/// Note: to be valid, constant expressions may not leave garbage data on the stack. It may leave only what is expected
14/// and nothing more.
15///
16/// Valid constant instructions are:
17/// - Core: <https://webassembly.github.io/spec/core/valid/instructions.html#valid-constant>
18/// - Extended Proposal: <https://webassembly.github.io/extended-const/core/valid/instructions.html#valid-constant>
19///
20/// # The Wonders of `global.get`
21/// The `global.get` instruction is quite picky by nature. To make a long story short, there are two rules to follow to
22/// be able to use this expression.
23///
24/// ## 1. The referenced global must be imported
25/// Take the example code:
26/// ```wat
27/// (module
28///     (global (export "g") (mut i32) (
29///         i32.add (i32.const 1) (i32.const 2)
30///     ))
31///
32///     (global (export "h1") i32 (
33///         i32.const 1
34///     ))
35///
36///     (global (export "h2") i32 (
37///         global.get 1
38///     ))
39///
40///     (func (export "f")
41///         i32.const 100
42///         global.set 0))
43/// ```
44///
45/// When compiling with wat2wasm, the following error is thrown:
46/// ```wat
47/// Error: validate failed:
48/// test.wast:11:24: error: initializer expression can only reference an imported global
49///             global.get 1
50///                        ^
51/// ```
52///
53/// When compiling the code with the latest dev build of wasmtime, the following error is thrown:
54/// ```wat
55/// failed to parse WebAssembly module
56///
57/// Caused by:
58///     constant expression required: global.get of locally defined global (at offset 0x24)
59/// ```
60///
61/// ## 2. The referenced global must be immutable
62///
63///```wat
64/// (module
65///     (import "env" "g" (global (mut i32)))
66///     (global (export "h") (mut i32) (
67///         i32.add (i32.const 1) (global.get 0)
68///     ))
69///   )
70/// ```
71///
72/// When compiling with wat2wasm, the following error is thrown:
73/// ```wat
74/// Error: validate failed:
75/// test.wast:4:27: error: initializer expression cannot reference a mutable global
76///     i32.add (i32.const 1) (global.get 0)
77/// ```
78///
79/// # Note
80/// The following instructions are not yet supported:
81/// - `ref.null`
82/// - `ref.func`
83/// - `global.get`
84pub fn read_constant_expression(
85    wasm: &mut WasmReader,
86    stack: &mut ValidationStack,
87    this_global_valtype: Option<ValType>,
88    _globals_ty: Option<&[GlobalType]>,
89    funcs: Option<&[usize]>,
90) -> Result<Span> {
91    let start_pc = wasm.pc;
92
93    loop {
94        let Ok(first_instr_byte) = wasm.read_u8() else {
95            return Err(Error::ExprMissingEnd);
96        };
97        trace!("Read constant instruction byte {first_instr_byte:#X?} ({first_instr_byte})");
98
99        use crate::core::reader::types::opcode::*;
100        match first_instr_byte {
101            // Missing: ref.null, ref.func, global.get
102            END => {
103                // The stack must only contain the global's valtype
104                if this_global_valtype.is_some() {
105                    stack.assert_val_types(&[this_global_valtype.unwrap()])?;
106                }
107                return Ok(Span::new(start_pc, wasm.pc - start_pc));
108            }
109            I32_CONST => {
110                let _num = wasm.read_var_i32()?;
111                stack.push_valtype(ValType::NumType(NumType::I32));
112            }
113            F32_CONST => {
114                let _num = wasm.read_var_f32();
115                stack.push_valtype(ValType::NumType(NumType::F32));
116            }
117            F64_CONST => {
118                let _num = wasm.read_var_f64();
119                stack.push_valtype(ValType::NumType(NumType::F64));
120            }
121            I64_CONST => {
122                let _num = wasm.read_var_i64()?;
123                stack.push_valtype(ValType::NumType(NumType::I64));
124            }
125            I32_ADD | I32_SUB | I32_MUL => {
126                stack.assert_pop_val_type(ValType::NumType(NumType::I32))?;
127                stack.assert_pop_val_type(ValType::NumType(NumType::I32))?;
128
129                stack.push_valtype(ValType::NumType(NumType::I32));
130            }
131            I64_ADD | I64_SUB | I64_MUL => {
132                stack.assert_pop_val_type(ValType::NumType(NumType::I64))?;
133                stack.assert_pop_val_type(ValType::NumType(NumType::I64))?;
134
135                stack.push_valtype(ValType::NumType(NumType::I64));
136            }
137            REF_NULL => {
138                stack.push_valtype(ValType::RefType(RefType::read(wasm)?));
139            }
140            REF_FUNC => {
141                let func_idx = wasm.read_var_u32()? as usize;
142                match funcs {
143                    Some(funcs) => {
144                        if func_idx >= funcs.len() {
145                            return Err(Error::FunctionIsNotDefined(func_idx));
146                        }
147                    }
148                    None => {
149                        return Err(Error::FunctionIsNotDefined(u32::MAX as usize));
150                    }
151                }
152
153                stack.push_valtype(ValType::RefType(crate::RefType::FuncRef));
154            }
155            _ => return Err(Error::InvalidInstr(first_instr_byte)),
156        }
157    }
158}
