29/// Information collected from validating a module.
30///
31/// This can be used to instantiate a new module instance in some
32/// [`Store`](crate::Store) thorugh
33/// [`Store::module_instantiate`](crate::Store::module_instantiate)
34#[derive(Clone, Debug)]
35pub struct ValidationInfo<'bytecode> {
36    pub(crate) wasm: &'bytecode [u8],
37    pub(crate) types: IdxVec<TypeIdx, FuncType>,
38    pub(crate) imports: Vec<Import<'bytecode>>,
39    pub(crate) functions: ExtendedIdxVec<FuncIdx, TypeIdx>,
40    pub(crate) tables: ExtendedIdxVec<TableIdx, TableType>,
41    pub(crate) memories: ExtendedIdxVec<MemIdx, MemType>,
42    pub(crate) globals: ExtendedIdxVec<GlobalIdx, Global>,
43    pub(crate) exports: Vec<Export<'bytecode>>,
44    pub(crate) elements: IdxVec<ElemIdx, ElemType>,
45    pub(crate) data: IdxVec<DataIdx, DataSegment>,
46    /// Each block contains the validated code section and the stp corresponding to
47    /// the beginning of that code section
48    pub(crate) func_blocks_stps: Vec<(Span, usize)>,
49    pub(crate) sidetable: Sidetable,
50    /// The start function which is automatically executed during instantiation
51    pub(crate) start: Option<FuncIdx>,
52    // pub(crate) exports_length: Exported,
53}

55fn validate_no_duplicate_exports(validation_info: &ValidationInfo) -> Result<(), ValidationError> {
56    let mut found_export_names: btree_set::BTreeSet<&str> = btree_set::BTreeSet::new();
57    for export in &validation_info.exports {
58        if found_export_names.contains(export.name) {
59            return Err(ValidationError::DuplicateExportName);
60        }
61        found_export_names.insert(export.name);
62    }
63    Ok(())
64}

66pub fn validate(wasm: &[u8]) -> Result<ValidationInfo<'_>, ValidationError> {
67    let mut wasm = WasmReader::new(wasm);
68
69    // represents C.refs in https://webassembly.github.io/spec/core/valid/conventions.html#context
70    // A func.ref instruction is onlv valid if it has an immediate that is a member of C.refs.
71    // this list holds all the func_idx's occurring in the module, except in its functions or start function.
72    // I make an exception here by not including func_idx's occuring within data segments in C.refs as well, so that single pass validation is possible.
73    // If there is a func_idx within the data segment, this would ultimately mean that data segment cannot be validated,
74    // therefore this hack is acceptable.
75    // https://webassembly.github.io/spec/core/valid/modules.html#data-segments
76    // https://webassembly.github.io/spec/core/valid/modules.html#valid-module
77
78    let mut validation_context_refs: BTreeSet<FuncIdx> = BTreeSet::new();
79
80    trace!("Starting validation of bytecode");
81
82    trace!("Validating magic value");
83    let [0x00, 0x61, 0x73, 0x6d] = wasm.strip_bytes::<4>()? else {
84        return Err(ValidationError::InvalidMagic);
85    };
86
87    trace!("Validating version number");
88    let [0x01, 0x00, 0x00, 0x00] = wasm.strip_bytes::<4>()? else {
89        return Err(ValidationError::InvalidBinaryFormatVersion);
90    };
91    debug!("Header ok");
92
93    let mut header = None;
94    read_next_header(&mut wasm, &mut header)?;
95
96    let skip_section = |wasm: &mut WasmReader, section_header: &mut Option<SectionHeader>| {
97        handle_section(wasm, section_header, SectionTy::Custom, |wasm, h| {
98            // customsec ::= section_0(custom)
99            // custom ::= name byte*
100            // name ::= b*:vec(byte) => name (if utf8(name) = b*)
101            // vec(B) ::= n:u32 (x:B)^n => x^n
102            let _name = wasm.read_name()?;
103
104            let remaining_bytes = h
105                .contents
106                .from()
107                .checked_add(h.contents.len())
108                .and_then(|res| res.checked_sub(wasm.pc))
109                .ok_or(ValidationError::InvalidCustomSectionLength)?;
110
111            // TODO: maybe do something with these remaining bytes?
112            let mut _bytes = Vec::new();
113            for _ in 0..remaining_bytes {
114                _bytes.push(wasm.read_u8()?)
115            }
116            Ok(())
117        })
118    };
119
120    while (skip_section(&mut wasm, &mut header)?).is_some() {}
121
122    let types = handle_section(&mut wasm, &mut header, SectionTy::Type, |wasm, _| {
123        wasm.read_vec(FuncType::read).map(|types| IdxVec::new(types).expect("that index space creation never fails because the length of the types vector is encoded as a 32-bit integer in the bytecode"))
124    })?
125    .unwrap_or_default();
126
127    while (skip_section(&mut wasm, &mut header)?).is_some() {}
128
129    let imports = handle_section(&mut wasm, &mut header, SectionTy::Import, |wasm, _| {
130        wasm.read_vec(|wasm| Import::read_and_validate(wasm, &types))
131    })?
132    .unwrap_or_default();
133
134    while (skip_section(&mut wasm, &mut header)?).is_some() {}
135
136    // The `Function` section only covers module-level (or "local") functions.
137    // Imported functions have their types known in the `import` section. Both
138    // local and imported functions share the same index space.
139    //
140    // Imported functions are given priority and have the first indicies, and
141    // only after that do the local functions get assigned their indices.
142    let local_functions =
143        handle_section(&mut wasm, &mut header, SectionTy::Function, |wasm, _| {
144            wasm.read_vec(|wasm| TypeIdx::read_and_validate(wasm, &types))
145        })?
146        .unwrap_or_default();
147
148    let imported_functions = imports.iter().filter_map(|import| match &import.desc {
149        ImportDesc::Func(type_idx) => Some(*type_idx),
150        _ => None,
151    });
152
153    let functions = ExtendedIdxVec::new(imported_functions.collect(), local_functions)
154        .map_err(|IdxVecOverflowError| ValidationError::TooManyFunctions)?;
155
156    while (skip_section(&mut wasm, &mut header)?).is_some() {}
157
158    let imported_tables = imports.iter().filter_map(|m| match m.desc {
159        ImportDesc::Table(table) => Some(table),
160        _ => None,
161    });
162    let local_tables = handle_section(&mut wasm, &mut header, SectionTy::Table, |wasm, _| {
163        wasm.read_vec(TableType::read)
164    })?
165    .unwrap_or_default();
166
167    let tables = ExtendedIdxVec::new(imported_tables.collect(), local_tables)
168        .map_err(|IdxVecOverflowError| ValidationError::TooManyTables)?;
169
170    while (skip_section(&mut wasm, &mut header)?).is_some() {}
171
172    let imported_memories = imports.iter().filter_map(|m| match m.desc {
173        ImportDesc::Mem(mem) => Some(mem),
174        _ => None,
175    });
176    // let imported_memories_length = imported_memories.len();
177    let local_memories = handle_section(&mut wasm, &mut header, SectionTy::Memory, |wasm, _| {
178        wasm.read_vec(MemType::read)
179    })?
180    .unwrap_or_default();
181
182    let memories = ExtendedIdxVec::new(imported_memories.collect(), local_memories)
183        .map_err(|IdxVecOverflowError| ValidationError::TooManyMemories)?;
184
185    if memories.inner().len() > 1 {
186        return Err(ValidationError::UnsupportedMultipleMemoriesProposal);
187    }
188
189    while (skip_section(&mut wasm, &mut header)?).is_some() {}
190
191    let imported_global_types: Vec<GlobalType> = imports
192        .iter()
193        .filter_map(|m| match m.desc {
194            ImportDesc::Global(global) => Some(global),
195            _ => None,
196        })
197        .collect();
198    let local_globals = handle_section(&mut wasm, &mut header, SectionTy::Global, |wasm, h| {
199        globals::validate_global_section(
200            wasm,
201            h,
202            &imported_global_types,
203            &mut validation_context_refs,
204            functions.inner(),
205        )
206    })?
207    .unwrap_or_default();
208
209    let imported_globals = imported_global_types.iter().map(|ty| Global {
210        // TODO using a default MAX value for spans that are never executed is
211        // not really safe. Maybe opt for an Option instead.
212        init_expr: Span::new(usize::MAX, 0),
213        ty: *ty,
214    });
215    let globals = ExtendedIdxVec::new(imported_globals.collect(), local_globals)
216        .map_err(|IdxVecOverflowError| ValidationError::TooManyGlobals)?;
217
218    while (skip_section(&mut wasm, &mut header)?).is_some() {}
219
220    let exports = handle_section(&mut wasm, &mut header, SectionTy::Export, |wasm, _| {
221        wasm.read_vec(|wasm| {
222            Export::read_and_validate(
223                wasm,
224                functions.inner(),
225                tables.inner(),
226                memories.inner(),
227                globals.inner(),
228            )
229        })
230    })?
231    .unwrap_or_default();
232    validation_context_refs.extend(exports.iter().filter_map(
233        |Export { name: _, desc }| match *desc {
234            ExportDesc::Func(func_idx) => Some(func_idx),
235            _ => None,
236        },
237    ));
238
239    while (skip_section(&mut wasm, &mut header)?).is_some() {}
240
241    let start = handle_section(&mut wasm, &mut header, SectionTy::Start, |wasm, _| {
242        let func_idx = FuncIdx::read_and_validate(wasm, functions.inner())?;
243
244        // start function signature must be [] -> []
245        // https://webassembly.github.io/spec/core/valid/modules.html#start-function
246        // SAFETY: We just validated this function index using the same
247        // `IdxVec`.
248        let type_idx = unsafe { functions.inner().get(func_idx) };
249
250        // SAFETY: There exists only one `IdxVec<TypeIdx, FuncType>` in the
251        // current function. Therefore, this has to be the same one used to
252        // create and validate this `TypeIdx`.
253        let func_type = unsafe { types.get(*type_idx) };
254        if func_type
255            != &(FuncType {
256                params: ResultType {
257                    valtypes: Vec::new(),
258                },
259                returns: ResultType {
260                    valtypes: Vec::new(),
261                },
262            })
263        {
264            Err(ValidationError::InvalidStartFunctionSignature)
265        } else {
266            Ok(func_idx)
267        }
268    })?;
269
270    while (skip_section(&mut wasm, &mut header)?).is_some() {}
271
272    let elements = handle_section(&mut wasm, &mut header, SectionTy::Element, |wasm, _| {
273        ElemType::read_and_validate(
274            wasm,
275            functions.inner(),
276            &mut validation_context_refs,
277            tables.inner(),
278            &imported_global_types,
279        )
280        .map(|elements| IdxVec::new(elements).expect("that index space creation never fails because the length of the elements vector is encoded as a 32-bit integer in the bytecode"))
281    })?
282    .unwrap_or_default();
283
284    while (skip_section(&mut wasm, &mut header)?).is_some() {}
285
286    // https://webassembly.github.io/spec/core/binary/modules.html#data-count-section
287    // As per the official documentation:
288    //
289    // The data count section is used to simplify single-pass validation. Since the data section occurs after the code section, the `memory.init` and `data.drop` and instructions would not be able to check whether the data segment index is valid until the data section is read. The data count section occurs before the code section, so a single-pass validator can use this count instead of deferring validation.
290    let data_count: Option<u32> =
291        handle_section(&mut wasm, &mut header, SectionTy::DataCount, |wasm, _| {
292            wasm.read_var_u32()
293        })?;
294    if let Some(dc) = data_count {
295        trace!("data count: {dc}");
296    }
297
298    while (skip_section(&mut wasm, &mut header)?).is_some() {}
299
300    let mut sidetable = Sidetable::new();
301    let func_blocks_stps = handle_section(&mut wasm, &mut header, SectionTy::Code, |wasm, h| {
302        // SAFETY: It is required that all passed index values are valid in all
303        // passed `IdxVec`s. The current function does not take any index types
304        // as arguments and every `IdxVec<..., ...>` is unique because they use
305        // different generics. Therefore, all index types must be valid in their
306        // relevant `IdxVec`s.
307        unsafe {
308            code::validate_code_section(
309                wasm,
310                h,
311                &types,
312                &functions,
313                globals.inner(),
314                memories.inner(),
315                data_count,
316                tables.inner(),
317                &elements,
318                &validation_context_refs,
319                &mut sidetable,
320            )
321        }
322    })?
323    .unwrap_or_default();
324
325    if func_blocks_stps.len() != functions.len_local_definitions().into_usize() {
326        return Err(ValidationError::FunctionAndCodeSectionsHaveDifferentLengths);
327    }
328
329    while (skip_section(&mut wasm, &mut header)?).is_some() {}
330
331    let data_section = handle_section(&mut wasm, &mut header, SectionTy::Data, |wasm, h| {
332        // wasm.read_vec(DataSegment::read)
333        data::validate_data_section(wasm, h, &imported_global_types, functions.inner(), memories.inner())
334            .map(|data_segments| IdxVec::new(data_segments).expect("that index space creation never fails because the length of the data segments vector is encoded as a 32-bit integer in the bytecode"))
335    })?
336    .unwrap_or_default();
337
338    // https://webassembly.github.io/spec/core/binary/modules.html#data-count-section
339    if let Some(data_count) = data_count {
340        if data_count != data_section.len() {
341            return Err(ValidationError::DataCountAndDataSectionsLengthAreDifferent);
342        }
343    }
344
345    while (skip_section(&mut wasm, &mut header)?).is_some() {}
346
347    // All sections should have been handled
348    if let Some(header) = header {
349        return Err(ValidationError::SectionOutOfOrder(header.ty));
350    }
351
352    debug!("Validation was successful");
353    let validation_info = ValidationInfo {
354        wasm: wasm.into_inner(),
355        types,
356        imports,
357        functions,
358        tables,
359        memories,
360        globals,
361        exports,
362        func_blocks_stps,
363        sidetable,
364        data: data_section,
365        start,
366        elements,
367    };
368    validate_no_duplicate_exports(&validation_info)?;
369
370    Ok(validation_info)
371}

373fn read_next_header(
374    wasm: &mut WasmReader,
375    header: &mut Option<SectionHeader>,
376) -> Result<(), ValidationError> {
377    if header.is_none() && !wasm.remaining_bytes().is_empty() {
378        *header = Some(SectionHeader::read(wasm)?);
379    }
380    Ok(())
381}

383#[inline(always)]
384fn handle_section<'wasm, T, F>(
385    wasm: &mut WasmReader<'wasm>,
386    header: &mut Option<SectionHeader>,
387    section_ty: SectionTy,
388    handler: F,
389) -> Result<Option<T>, ValidationError>
390where
391    T: 'wasm,
392    F: FnOnce(&mut WasmReader<'wasm>, SectionHeader) -> Result<T, ValidationError>,
393{
394    match &header {
395        Some(SectionHeader { ty, .. }) if *ty == section_ty => {
396            let h = header.take().unwrap();
397            trace!("Handling section {:?}", h.ty);
398            let ret = handler(wasm, h)?;
399            read_next_header(wasm, header)?;
400            Ok(Some(ret))
401        }
402        _ => Ok(None),
403    }
404}

406impl<'wasm> ValidationInfo<'wasm> {
407    /// Returns the imports of this module as an iterator. Each import consist
408    /// of a module name, a name and an extern type.
409    ///
410    /// See: WebAssembly Specification 2.0 - 7.1.5 - module_imports
411    pub fn imports<'a>(
412        &'a self,
413    ) -> Map<
414        core::slice::Iter<'a, Import<'wasm>>,
415        impl FnMut(&'a Import<'wasm>) -> (&'a str, &'a str, ExternType),
416    > {
417        self.imports.iter().map(|import| {
418            // SAFETY: This is sound because the argument is `self` and the
419            // import desc also comes from `self`.
420            let extern_type = unsafe { import.desc.extern_type(self) };
421            (import.module_name, import.name, extern_type)
422        })
423    }
424
425    /// Returns the exports of this module as an iterator. Each export consist
426    /// of a name, and an extern type.
427    ///
428    /// See: WebAssembly Specification 2.0 - 7.1.5 - module_exports
429    pub fn exports<'a>(
430        &'a self,
431    ) -> Map<
432        core::slice::Iter<'a, Export<'wasm>>,
433        impl FnMut(&'a Export<'wasm>) -> (&'a str, ExternType),
434    > {
435        self.exports.iter().map(|export| {
436            // SAFETY: This is sound because the argument is `self` and the
437            // export desc also comes from `self`.
438            let extern_type = unsafe { export.desc.extern_type(self) };
439            (export.name, extern_type)
440        })
441    }
442}
