22/// Information collected from validating a module.
23/// This can be used to create a [crate::RuntimeInstance].
24pub struct ValidationInfo<'bytecode> {
25    pub(crate) wasm: &'bytecode [u8],
26    pub(crate) types: Vec<FuncType>,
27    pub(crate) imports: Vec<Import>,
28    pub(crate) functions: Vec<TypeIdx>,
29    pub(crate) tables: Vec<TableType>,
30    pub(crate) memories: Vec<MemType>,
31    pub(crate) globals: Vec<Global>,
32    #[allow(dead_code)]
33    pub(crate) exports: Vec<Export>,
34    /// Each block contains the validated code section and the generated sidetable
35    pub(crate) func_blocks: Vec<(Span, Sidetable)>,
36    pub(crate) data: Vec<DataSegment>,
37    /// The start function which is automatically executed during instantiation
38    pub(crate) start: Option<FuncIdx>,
39    pub(crate) elements: Vec<ElemType>,
40}

42pub fn validate(wasm: &[u8]) -> Result<ValidationInfo> {
43    let mut wasm = WasmReader::new(wasm);
44    trace!("Starting validation of bytecode");
45
46    trace!("Validating magic value");
47    let [0x00, 0x61, 0x73, 0x6d] = wasm.strip_bytes::<4>()? else {
48        return Err(Error::InvalidMagic);
49    };
50
51    trace!("Validating version number");
52    let [0x01, 0x00, 0x00, 0x00] = wasm.strip_bytes::<4>()? else {
53        return Err(Error::InvalidVersion);
54    };
55    debug!("Header ok");
56
57    let mut header = None;
58    read_next_header(&mut wasm, &mut header)?;
59
60    let skip_section = |wasm: &mut WasmReader, section_header: &mut Option<SectionHeader>| {
61        handle_section(wasm, section_header, SectionTy::Custom, |wasm, h| {
62            use alloc::string::*;
63            // customsec ::= section_0(custom)
64            // custom ::= name byte*
65            // name ::= b*:vec(byte) => name (if utf8(name) = b*)
66            // vec(B) ::= n:u32 (x:B)^n => x^n
67
68            if h.contents.len() == 0 {
69                return Ok(());
70            }
71            let _name = wasm.read_name()?;
72
73            let remaining_bytes = match h
74                .contents
75                .from()
76                .checked_add(h.contents.len())
77                .and_then(|res| res.checked_sub(wasm.pc))
78            {
79                None => Err(Error::InvalidSection(
80                    SectionTy::Custom,
81                    "Remaining bytes less than 0 after reading name!".to_string(),
82                )),
83                Some(remaining_bytes) => Ok(remaining_bytes),
84            }?;
85
86            // TODO: maybe do something with these remaining bytes?
87            let mut _bytes = Vec::new();
88            for _ in 0..remaining_bytes {
89                _bytes.push(wasm.read_u8()?)
90            }
91            Ok(())
92        })
93    };
94
95    while (skip_section(&mut wasm, &mut header)?).is_some() {}
96
97    let types = handle_section(&mut wasm, &mut header, SectionTy::Type, |wasm, _| {
98        wasm.read_vec(FuncType::read)
99    })?
100    .unwrap_or_default();
101
102    while (skip_section(&mut wasm, &mut header)?).is_some() {}
103
104    let imports = handle_section(&mut wasm, &mut header, SectionTy::Import, |wasm, _| {
105        wasm.read_vec(Import::read)
106    })?
107    .unwrap_or_default();
108
109    while (skip_section(&mut wasm, &mut header)?).is_some() {}
110
111    // The `Function` section only covers module-level (or "local") functions.
112    // Imported functions have their types known in the `import` section. Both
113    // local and imported functions share the same index space.
114    //
115    // Imported functions are given priority and have the first indicies, and
116    // only after that do the local functions get assigned their indices.
117    let local_functions =
118        handle_section(&mut wasm, &mut header, SectionTy::Function, |wasm, _| {
119            wasm.read_vec(|wasm| wasm.read_var_u32().map(|u| u as usize))
120        })?
121        .unwrap_or_default();
122
123    let imported_functions = imports.iter().filter_map(|import| match &import.desc {
124        ImportDesc::Func(type_idx) => Some(*type_idx),
125        _ => None,
126    });
127
128    let all_functions = imported_functions
129        .clone()
130        .chain(local_functions.iter().cloned())
131        .collect::<Vec<TypeIdx>>();
132
133    while (skip_section(&mut wasm, &mut header)?).is_some() {}
134
135    let tables = handle_section(&mut wasm, &mut header, SectionTy::Table, |wasm, _| {
136        wasm.read_vec(TableType::read)
137    })?
138    .unwrap_or_default();
139
140    while (skip_section(&mut wasm, &mut header)?).is_some() {}
141
142    let memories = handle_section(&mut wasm, &mut header, SectionTy::Memory, |wasm, _| {
143        wasm.read_vec(MemType::read)
144    })?
145    .unwrap_or_default();
146    if memories.len() > 1 {
147        return Err(Error::MoreThanOneMemory);
148    }
149
150    while (skip_section(&mut wasm, &mut header)?).is_some() {}
151
152    let globals = handle_section(&mut wasm, &mut header, SectionTy::Global, |wasm, h| {
153        globals::validate_global_section(wasm, h)
154    })?
155    .unwrap_or_default();
156
157    while (skip_section(&mut wasm, &mut header)?).is_some() {}
158
159    let exports = handle_section(&mut wasm, &mut header, SectionTy::Export, |wasm, _| {
160        wasm.read_vec(Export::read)
161    })?
162    .unwrap_or_default();
163
164    while (skip_section(&mut wasm, &mut header)?).is_some() {}
165
166    let start = handle_section(&mut wasm, &mut header, SectionTy::Start, |wasm, _| {
167        wasm.read_var_u32().map(|idx| idx as FuncIdx)
168    })?;
169
170    while (skip_section(&mut wasm, &mut header)?).is_some() {}
171
172    let mut referenced_functions = btree_set::BTreeSet::new();
173    let elements: Vec<ElemType> =
174        handle_section(&mut wasm, &mut header, SectionTy::Element, |wasm, _| {
175            ElemType::read_from_wasm(
176                wasm,
177                &all_functions,
178                &mut referenced_functions,
179                tables.len(),
180            )
181        })?
182        .unwrap_or_default();
183
184    while (skip_section(&mut wasm, &mut header)?).is_some() {}
185
186    // https://webassembly.github.io/spec/core/binary/modules.html#data-count-section
187    // As per the official documentation:
188    //
189    // The data count section is used to simplify single-pass validation. Since the data section occurs after the code section, the `memory.init` and `data.drop` and instructions would not be able to check whether the data segment index is valid until the data section is read. The data count section occurs before the code section, so a single-pass validator can use this count instead of deferring validation.
190    let data_count: Option<u32> =
191        handle_section(&mut wasm, &mut header, SectionTy::DataCount, |wasm, _| {
192            wasm.read_var_u32()
193        })?;
194    if data_count.is_some() {
195        trace!("data count: {}", data_count.unwrap());
196    }
197
198    while (skip_section(&mut wasm, &mut header)?).is_some() {}
199
200    let func_blocks_sidetables =
201        handle_section(&mut wasm, &mut header, SectionTy::Code, |wasm, h| {
202            code::validate_code_section(
203                wasm,
204                h,
205                &types,
206                &all_functions,
207                imported_functions.count(),
208                &globals,
209                &memories,
210                &data_count,
211                &tables,
212                &elements,
213                &referenced_functions,
214            )
215        })?
216        .unwrap_or_default();
217
218    assert_eq!(
219        func_blocks_sidetables.len(),
220        local_functions.len(),
221        "these should be equal"
222    ); // TODO check if this is in the spec
223
224    while (skip_section(&mut wasm, &mut header)?).is_some() {}
225
226    let data_section = handle_section(&mut wasm, &mut header, SectionTy::Data, |wasm, _| {
227        wasm.read_vec(DataSegment::read)
228    })?
229    .unwrap_or_default();
230
231    // https://webassembly.github.io/spec/core/binary/modules.html#data-count-section
232    if data_count.is_some() {
233        assert_eq!(data_count.unwrap() as usize, data_section.len());
234    }
235
236    while (skip_section(&mut wasm, &mut header)?).is_some() {}
237
238    // All sections should have been handled
239    if let Some(header) = header {
240        return Err(Error::SectionOutOfOrder(header.ty));
241    }
242
243    debug!("Validation was successful");
244    Ok(ValidationInfo {
245        wasm: wasm.into_inner(),
246        types,
247        imports,
248        functions: local_functions,
249        tables,
250        memories,
251        globals,
252        exports,
253        func_blocks: func_blocks_sidetables,
254        data: data_section,
255        start,
256        elements,
257    })
258}

260fn read_next_header(wasm: &mut WasmReader, header: &mut Option<SectionHeader>) -> Result<()> {
261    if header.is_none() && !wasm.remaining_bytes().is_empty() {
262        *header = Some(SectionHeader::read(wasm)?);
263    }
264    Ok(())
265}

267#[inline(always)]
268fn handle_section<T, F: FnOnce(&mut WasmReader, SectionHeader) -> Result<T>>(
269    wasm: &mut WasmReader,
270    header: &mut Option<SectionHeader>,
271    section_ty: SectionTy,
272    handler: F,
273) -> Result<Option<T>> {
274    match &header {
275        Some(SectionHeader { ty, .. }) if *ty == section_ty => {
276            let h = header.take().unwrap();
277            trace!("Handling section {:?}", h.ty);
278            let ret = handler(wasm, h)?;
279            read_next_header(wasm, header)?;
280            Ok(Some(ret))
281        }
282        _ => Ok(None),
283    }
